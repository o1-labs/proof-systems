1use arkworks::{WasmPastaFp, WasmPastaFq};
2use mina_curves::pasta::{Fp, Fq};
3use mina_poseidon::{constants::PlonkSpongeConstantsKimchi, permutation::poseidon_block_cipher};
4use neon::{
5    context::FunctionContext,
6    result::JsResult,
7    types::{buffer::TypedArray, JsUint8Array},
8};
9use wasm_types::{FlatVector, FlatVectorElem};

11// fp
12
13pub fn caml_pasta_fp_poseidon_block_cipher(mut cx: FunctionContext) -> JsResult<JsUint8Array> {
14    let state: &JsUint8Array = &*cx.argument(0)?;
15
16    let mut state: Vec<Fp> = FlatVector::<WasmPastaFp>::from_bytes(state.as_slice(&cx).to_vec())
17        .into_iter()
18        .map(Into::into)
19        .collect();
20
21    poseidon_block_cipher::<Fp, PlonkSpongeConstantsKimchi>(
22        mina_poseidon::pasta::fp_kimchi::static_params(),
23        &mut state,
24    );
25
26    let res = state
27        .iter()
28        .map(|f| WasmPastaFp(*f))
29        .flat_map(FlatVectorElem::flatten)
30        .collect::<Vec<u8>>();
31
32    JsUint8Array::from_slice(&mut cx, &res)
33}
