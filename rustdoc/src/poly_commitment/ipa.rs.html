64/// Computes the endomorphism coefficients (ξ, λ) for a curve.
65///
66/// For curves of the form y² = x³ + b (like Pallas and Vesta), there exists
67/// an efficient endomorphism φ defined by:
68///
69///   φ(x, y) = (ξ · x, y)
70///
71/// where ξ is a primitive cube root of unity in the base field (ξ³ = 1, ξ ≠ 1).
72/// This works because (ξx)³ = ξ³x³ = x³, so the point remains on the curve.
73///
74/// This endomorphism corresponds to scalar multiplication by λ:
75///
76///   φ(P) = [λ]P
77///
78/// where λ is a primitive cube root of unity in the scalar field.
79///
80/// # Returns
81///
82/// A tuple `(endo_q, endo_r)` where:
83/// - `endo_q` (ξ): cube root of unity in the base field F_q, used to compute φ(P)
84/// - `endo_r` (λ): the corresponding scalar in F_r such that φ(P) = [λ]P
85///
86/// # Mathematical Background
87///
88/// The cube root is computed as ξ = g^((p-1)/3) where g is a generator of F_p*.
89/// By Fermat's Little Theorem, ξ³ = g^(p-1) = 1.
90///
91/// Since there are two primitive cube roots of unity (ξ and ξ²), the function
92/// verifies which one corresponds to the endomorphism by checking:
93///
94///   [potential_λ]G == φ(G)
95///
96/// If not, it uses λ = potential_λ² instead.
97///
98/// # References
99///
100/// - Halo paper, Section 6.2: <https://eprint.iacr.org/2019/1021>
101/// - GLV method for fast scalar multiplication
102pub fn endos<G: CommitmentCurve>() -> (G::BaseField, G::ScalarField)
103where
104    G::BaseField: PrimeField,
105{
106    let endo_q: G::BaseField = mina_poseidon::sponge::endo_coefficient();
107    let endo_r = {
108        let potential_endo_r: G::ScalarField = mina_poseidon::sponge::endo_coefficient();
109        let t = G::generator();
110        let (x, y) = t.to_coordinates().unwrap();
111        let phi_t = G::of_coordinates(x * endo_q, y);
112        if t.mul(potential_endo_r) == phi_t.into_group() {
113            potential_endo_r
114        } else {
115            potential_endo_r * potential_endo_r
116        }
117    };
118    (endo_q, endo_r)
119}

121fn point_of_random_bytes<G: CommitmentCurve>(map: &G::Map, random_bytes: &[u8]) -> G
122where
123    G::BaseField: Field,
124{
125    // packing in bit-representation
126    const N: usize = 31;
127    let extension_degree = G::BaseField::extension_degree() as usize;
128
129    let mut base_fields = Vec::with_capacity(N * extension_degree);
130
131    for base_count in 0..extension_degree {
132        let mut bits = [false; 8 * N];
133        let offset = base_count * N;
134        for i in 0..N {
135            for j in 0..8 {
136                bits[8 * i + j] = (random_bytes[offset + i] >> j) & 1 == 1;
137            }
138        }
139
140        let n =
141            <<G::BaseField as Field>::BasePrimeField as PrimeField>::BigInt::from_bits_be(&bits);
142        let t = <<G::BaseField as Field>::BasePrimeField as PrimeField>::from_bigint(n)
143            .expect("packing code has a bug");
144        base_fields.push(t)
145    }
146
147    let t = G::BaseField::from_base_prime_field_elems(base_fields).unwrap();
148
149    let (x, y) = map.to_group(t);
150    G::of_coordinates(x, y).mul_by_cofactor()
151}

155    /// This function verifies a batch of polynomial commitment opening proofs.
156    /// Return `true` if the verification is successful, `false` otherwise.
157    pub fn verify<EFqSponge, RNG, const FULL_ROUNDS: usize>(
158        &self,
159        group_map: &G::Map,
160        batch: &mut [BatchEvaluationProof<
161            G,
162            EFqSponge,
163            OpeningProof<G, FULL_ROUNDS>,
164            FULL_ROUNDS,
165        >],
166        rng: &mut RNG,
167    ) -> bool
168    where
169        EFqSponge: FqSponge<G::BaseField, G, G::ScalarField, FULL_ROUNDS>,
170        RNG: RngCore + CryptoRng,
171        G::BaseField: PrimeField,
172    {
173        // Verifier checks for all i,
174        // c_i Q_i + delta_i = z1_i (G_i + b_i U_i) + z2_i H
175        //
176        // if we sample evalscale at random, it suffices to check
177        //
178        // 0 == sum_i evalscale^i (c_i Q_i + delta_i - ( z1_i (G_i + b_i U_i) + z2_i H ))
179        //
180        // and because each G_i is a multiexp on the same array self.g, we
181        // can batch the multiexp across proofs.
182        //
183        // So for each proof in the batch, we add onto our big multiexp the
184        // following terms
185        // evalscale^i c_i Q_i
186        // evalscale^i delta_i
187        // - (evalscale^i z1_i) G_i
188        // - (evalscale^i z2_i) H
189        // - (evalscale^i z1_i b_i) U_i
190
191        // We also check that the sg component of the proof is equal to the
192        // polynomial commitment to the "s" array
193
194        let nonzero_length = self.g.len();
195
196        let max_rounds = math::ceil_log2(nonzero_length);
197
198        let padded_length = 1 << max_rounds;
199
200        let (_, endo_r) = endos::<G>();
201
202        // TODO: This will need adjusting
203        let padding = padded_length - nonzero_length;
204        let mut points = vec![self.h];
205        points.extend(self.g.clone());
206        points.extend(vec![G::zero(); padding]);
207
208        let mut scalars = vec![G::ScalarField::zero(); padded_length + 1];
209        assert_eq!(scalars.len(), points.len());
210
211        // sample randomiser to scale the proofs with
212        let rand_base = G::ScalarField::rand(rng);
213        let sg_rand_base = G::ScalarField::rand(rng);
214
215        let mut rand_base_i = G::ScalarField::one();
216        let mut sg_rand_base_i = G::ScalarField::one();
217
218        for BatchEvaluationProof {
219            sponge,
220            evaluation_points,
221            polyscale,
222            evalscale,
223            evaluations,
224            opening,
225            combined_inner_product,
226        } in batch.iter_mut()
227        {
228            sponge.absorb_fr(&[shift_scalar::<G>(*combined_inner_product)]);
229
230            let u_base: G = {
231                let t = sponge.challenge_fq();
232                let (x, y) = group_map.to_group(t);
233                G::of_coordinates(x, y)
234            };
235
236            let Challenges { chal, chal_inv } = opening.challenges::<EFqSponge>(&endo_r, sponge);
237
238            sponge.absorb_g(&[opening.delta]);
239            let c = ScalarChallenge::new(sponge.challenge()).to_field(&endo_r);
240
241            // < s, sum_i evalscale^i pows(evaluation_point[i]) >
242            // ==
243            // sum_i evalscale^i < s, pows(evaluation_point[i]) >
244            let b0 = {
245                let mut scale = G::ScalarField::one();
246                let mut res = G::ScalarField::zero();
247                for &e in evaluation_points.iter() {
248                    let term = b_poly(&chal, e);
249                    res += &(scale * term);
250                    scale *= *evalscale;
251                }
252                res
253            };
254
255            let s = b_poly_coefficients(&chal);
256
257            let neg_rand_base_i = -rand_base_i;
258
259            // TERM
260            // - rand_base_i z1 G
261            //
262            // we also add -sg_rand_base_i * G to check correctness of sg.
263            points.push(opening.sg);
264            scalars.push(neg_rand_base_i * opening.z1 - sg_rand_base_i);
265
266            // Here we add
267            // sg_rand_base_i * ( < s, self.g > )
268            // =
269            // < sg_rand_base_i s, self.g >
270            //
271            // to check correctness of the sg component.
272            {
273                let terms: Vec<_> = s.par_iter().map(|s| sg_rand_base_i * s).collect();
274
275                for (i, term) in terms.iter().enumerate() {
276                    scalars[i + 1] += term;
277                }
278            }
279
280            // TERM
281            // - rand_base_i * z2 * H
282            scalars[0] -= &(rand_base_i * opening.z2);
283
284            // TERM
285            // -rand_base_i * (z1 * b0 * U)
286            scalars.push(neg_rand_base_i * (opening.z1 * b0));
287            points.push(u_base);
288
289            // TERM
290            // rand_base_i c_i Q_i
291            // = rand_base_i c_i
292            //   (sum_j (chal_invs[j] L_j + chals[j] R_j) + P_prime)
293            // where P_prime = combined commitment + combined_inner_product * U
294            let rand_base_i_c_i = c * rand_base_i;
295            for ((l, r), (u_inv, u)) in opening.lr.iter().zip(chal_inv.iter().zip(chal.iter())) {
296                points.push(*l);
297                scalars.push(rand_base_i_c_i * u_inv);
298
299                points.push(*r);
300                scalars.push(rand_base_i_c_i * u);
301            }
302
303            // TERM
304            // sum_j evalscale^j (sum_i polyscale^i f_i) (elm_j)
305            // == sum_j sum_i evalscale^j polyscale^i f_i(elm_j)
306            // == sum_i polyscale^i sum_j evalscale^j f_i(elm_j)
307            combine_commitments(
308                evaluations,
309                &mut scalars,
310                &mut points,
311                *polyscale,
312                rand_base_i_c_i,
313            );
314
315            scalars.push(rand_base_i_c_i * *combined_inner_product);
316            points.push(u_base);
317
318            scalars.push(rand_base_i);
319            points.push(opening.delta);
320
321            rand_base_i *= &rand_base;
322            sg_rand_base_i *= &sg_rand_base;
323        }
324
325        // Verify the equation in two chunks, which is optimal for our SRS size.
326        // (see the comment to the `benchmark_msm_parallel_vesta` MSM benchmark)
327        let chunk_size = points.len() / 2;
328        let msm_res = points
329            .into_par_iter()
330            .chunks(chunk_size)
331            .zip(scalars.into_par_iter().chunks(chunk_size))
332            .map(|(bases, coeffs)| {
333                let coeffs_bigint = coeffs
334                    .into_iter()
335                    .map(|c| c.into_bigint())
336                    .collect::<Vec<_>>();
337                G::Group::msm_bigint(&bases, &coeffs_bigint)
338            })
339            .reduce(G::Group::zero, |mut l, r| {
340                l += r;
341                l
342            });
343
344        msm_res == G::Group::zero()
345    }

347    /// This function creates a trusted-setup SRS instance for circuits with
348    /// number of rows up to `depth`.
349    ///
350    /// # Safety
351    ///
352    /// This function is unsafe because it creates a trusted setup and the toxic
353    /// waste is passed as a parameter.
354    pub unsafe fn create_trusted_setup(x: G::ScalarField, depth: usize) -> Self {
355        let m = G::Map::setup();
356
357        let mut x_pow = G::ScalarField::one();
358        let g: Vec<_> = (0..depth)
359            .map(|_| {
360                let res = G::generator().mul(x_pow);
361                x_pow *= x;
362                res.into_affine()
363            })
364            .collect();
365
366        // Compute a blinder
367        let h = {
368            let mut h = Blake2b512::new();
369            h.update("srs_misc".as_bytes());
370            // FIXME: This is for retrocompatibility with a previous version
371            // that was using a list initialisation. It is not necessary.
372            h.update(0_u32.to_be_bytes());
373            point_of_random_bytes(&m, &h.finalize())
374        };
375
376        Self {
377            g,
378            h,
379            lagrange_bases: HashMapCache::new(),
380        }
381    }
382}

434    /// Turns a non-hiding polynomial commitment into a hiding polynomial
435    /// commitment. Transforms each given `<a, G>` into `(<a, G> + wH, w)` with
436    /// a random `w` per commitment.
437    fn mask(
438        &self,
439        comm: PolyComm<G>,
440        rng: &mut (impl RngCore + CryptoRng),
441    ) -> BlindedCommitment<G> {
442        let blinders = comm.map(|_| G::ScalarField::rand(rng));
443        self.mask_custom(comm, &blinders).unwrap()
444    }
445
446    fn mask_custom(
447        &self,
448        com: PolyComm<G>,
449        blinders: &PolyComm<G::ScalarField>,
450    ) -> Result<BlindedCommitment<G>, CommitmentError> {
451        let commitment = com
452            .zip(blinders)
453            .ok_or_else(|| CommitmentError::BlindersDontMatch(blinders.len(), com.len()))?
454            .map(|(g, b)| {
455                let mut g_masked = self.h.mul(b);
456                g_masked.add_assign(&g);
457                g_masked.into_affine()
458            });
459        Ok(BlindedCommitment {
460            commitment,
461            blinders: blinders.clone(),
462        })
463    }
464
465    fn commit_non_hiding(
466        &self,
467        plnm: &DensePolynomial<G::ScalarField>,
468        num_chunks: usize,
469    ) -> PolyComm<G> {
470        let is_zero = plnm.is_zero();
471
472        // chunk while committing
473        let mut chunks: Vec<_> = if is_zero {
474            vec![G::zero()]
475        } else if plnm.len() < self.g.len() {
476            vec![G::Group::msm(&self.g[..plnm.len()], &plnm.coeffs)
477                .unwrap()
478                .into_affine()]
479        } else if plnm.len() == self.g.len() {
480            // when processing a single chunk, it's faster to parallelise
481            // vertically in 2 threads (see the comment to the
482            // `benchmark_msm_parallel_vesta` MSM benchmark)
483            let n = self.g.len();
484            let (r1, r2) = rayon::join(
485                || G::Group::msm(&self.g[..n / 2], &plnm.coeffs[..n / 2]).unwrap(),
486                || G::Group::msm(&self.g[n / 2..n], &plnm.coeffs[n / 2..n]).unwrap(),
487            );
488
489            vec![(r1 + r2).into_affine()]
490        } else {
491            // otherwise it's better to parallelise horizontally along chunks
492            plnm.into_par_iter()
493                .chunks(self.g.len())
494                .map(|chunk| {
495                    let chunk_coeffs = chunk
496                        .into_iter()
497                        .map(|c| c.into_bigint())
498                        .collect::<Vec<_>>();
499                    let chunk_res = G::Group::msm_bigint(&self.g, &chunk_coeffs);
500                    chunk_res.into_affine()
501                })
502                .collect()
503        };
504
505        for _ in chunks.len()..num_chunks {
506            chunks.push(G::zero());
507        }
508
509        PolyComm::<G>::new(chunks)
510    }
511
512    fn commit(
513        &self,
514        plnm: &DensePolynomial<G::ScalarField>,
515        num_chunks: usize,
516        rng: &mut (impl RngCore + CryptoRng),
517    ) -> BlindedCommitment<G> {
518        self.mask(self.commit_non_hiding(plnm, num_chunks), rng)
519    }
520
521    fn commit_custom(
522        &self,
523        plnm: &DensePolynomial<G::ScalarField>,
524        num_chunks: usize,
525        blinders: &PolyComm<G::ScalarField>,
526    ) -> Result<BlindedCommitment<G>, CommitmentError> {
527        self.mask_custom(self.commit_non_hiding(plnm, num_chunks), blinders)
528    }
529
530    fn commit_evaluations_non_hiding(
531        &self,
532        domain: D<G::ScalarField>,
533        plnm: &Evaluations<G::ScalarField, D<G::ScalarField>>,
534    ) -> PolyComm<G> {
535        let basis = self.get_lagrange_basis(domain);
536        let commit_evaluations = |evals: &Vec<G::ScalarField>, basis: &Vec<PolyComm<G>>| {
537            PolyComm::<G>::multi_scalar_mul(&basis.iter().collect::<Vec<_>>()[..], &evals[..])
538        };
539        match domain.size.cmp(&plnm.domain().size) {
540            std::cmp::Ordering::Less => {
541                let s = (plnm.domain().size / domain.size) as usize;
542                let v: Vec<_> = (0..(domain.size())).map(|i| plnm.evals[s * i]).collect();
543                commit_evaluations(&v, basis)
544            }
545            std::cmp::Ordering::Equal => commit_evaluations(&plnm.evals, basis),
546            std::cmp::Ordering::Greater => {
547                panic!("desired commitment domain size ({}) greater than evaluations' domain size ({}):", domain.size, plnm.domain().size)
548            }
549        }
550    }
551

553        &self,
554        domain: D<G::ScalarField>,
555        plnm: &Evaluations<G::ScalarField, D<G::ScalarField>>,
556        rng: &mut (impl RngCore + CryptoRng),
557    ) -> BlindedCommitment<G> {
558        self.mask(self.commit_evaluations_non_hiding(domain, plnm), rng)

614impl<G: CommitmentCurve> SRS<G> {
615    #[allow(clippy::type_complexity)]
616    #[allow(clippy::many_single_char_names)]
617    // NB: a slight modification to the original protocol is done when absorbing
618    // the first prover message to improve the efficiency in a recursive
619    // setting.
620    pub fn open<EFqSponge, RNG, D: EvaluationDomain<G::ScalarField>, const FULL_ROUNDS: usize>(
621        &self,
622        group_map: &G::Map,
623        plnms: PolynomialsToCombine<G, D>,
624        elm: &[G::ScalarField],
625        polyscale: G::ScalarField,
626        evalscale: G::ScalarField,
627        mut sponge: EFqSponge,
628        rng: &mut RNG,
629    ) -> OpeningProof<G, FULL_ROUNDS>
630    where
631        EFqSponge: Clone + FqSponge<G::BaseField, G, G::ScalarField, FULL_ROUNDS>,
632        RNG: RngCore + CryptoRng,
633        G::BaseField: PrimeField,
634        G: EndoCurve,
635    {
636        let (endo_q, endo_r) = endos::<G>();
637
638        let rounds = math::ceil_log2(self.g.len());
639        let padded_length = 1 << rounds;
640
641        // TODO: Trim this to the degree of the largest polynomial
642        // TODO: We do always suppose we have a power of 2 for the SRS in
643        // practice. Therefore, padding equals zero, and this code can be
644        // removed. Only a current test case uses a SRS with a non-power of 2.
645        let padding = padded_length - self.g.len();
646        let mut g = self.g.clone();
647        g.extend(vec![G::zero(); padding]);
648
649        // Combines polynomials roughly as follows: p(X) := ∑_i polyscale^i p_i(X)
650        //
651        // `blinding_factor` is a combined set of commitments that are
652        // paired with polynomials in `plnms`. In kimchi, these input
653        // commitments are poly com blinders, so often `[G::ScalarField::one();
654        // num_chunks]` or zeroes.
655        let (p, blinding_factor) = combine_polys::<G, D>(plnms, polyscale, self.g.len());
656
657        // The initial evaluation vector for polynomial commitment b_init is not
658        // just the powers of a single point as in the original IPA
659        // (1,ζ,ζ^2,...)
660        //
661        // but rather a vector of linearly combined powers with `evalscale` as
662        // recombiner.
663        //
664        // b_init[j] = Σ_i evalscale^i elm_i^j
665        //           = ζ^j + evalscale * ζ^j ω^j (in the specific case of challenges (ζ,ζω))
666        //
667        // So in our case b_init is the following vector:
668        //    1 + evalscale
669        //    ζ + evalscale * ζ ω
670        //    ζ^2 + evalscale * (ζ ω)^2
671        //    ζ^3 + evalscale * (ζ ω)^3
672        //    ...
673        let b_init = {
674            // randomise/scale the eval powers
675            let mut scale = G::ScalarField::one();
676            let mut res: Vec<G::ScalarField> =
677                (0..padded_length).map(|_| G::ScalarField::zero()).collect();
678            for e in elm {
679                for (i, t) in pows(padded_length, *e).iter().enumerate() {
680                    res[i] += &(scale * t);
681                }
682                scale *= &evalscale;
683            }
684            res
685        };
686
687        // Combined polynomial p(X) evaluated at the combined eval point b_init.
688        let combined_inner_product = p
689            .coeffs
690            .iter()
691            .zip(b_init.iter())
692            .map(|(a, b)| *a * b)
693            .fold(G::ScalarField::zero(), |acc, x| acc + x);
694
695        // Usually, the prover sends `combined_inner_product`` to the verifier
696        // So we should absorb `combined_inner_product``
697        // However it is more efficient in the recursion circuit
698        // to absorb a slightly modified version of it.
699        // As a reminder, in a recursive setting, the challenges are given as a
700        // public input and verified in the next iteration.
701        // See the `shift_scalar`` doc.
702        sponge.absorb_fr(&[shift_scalar::<G>(combined_inner_product)]);
703
704        // Generate another randomisation base U; our commitments will be w.r.t
705        // bases {G_i},H,U.
706        let u_base: G = {
707            let t = sponge.challenge_fq();
708            let (x, y) = group_map.to_group(t);
709            G::of_coordinates(x, y)
710        };
711
712        let mut a = p.coeffs;
713        assert!(padded_length >= a.len());
714        a.extend(vec![G::ScalarField::zero(); padded_length - a.len()]);
715
716        let mut b = b_init;
717
718        let mut lr = vec![];
719
720        let mut blinders = vec![];
721
722        let mut chals = vec![];
723        let mut chal_invs = vec![];
724
725        // The main IPA folding loop that has log iterations.
726        for _ in 0..rounds {
727            let n = g.len() / 2;
728            // Pedersen bases
729            let (g_lo, g_hi) = (&g[0..n], &g[n..]);
730            // Polynomial coefficients
731            let (a_lo, a_hi) = (&a[0..n], &a[n..]);
732            // Evaluation points
733            let (b_lo, b_hi) = (&b[0..n], &b[n..]);
734
735            // Blinders for L/R
736            let rand_l = <G::ScalarField as UniformRand>::rand(rng);
737            let rand_r = <G::ScalarField as UniformRand>::rand(rng);
738
739            // Pedersen commitment to a_lo,rand_l,<a_hi,b_lo>
740            let l = G::Group::msm_bigint(
741                &[g_lo, &[self.h, u_base]].concat(),
742                &[a_hi, &[rand_l, inner_prod(a_hi, b_lo)]]
743                    .concat()
744                    .iter()
745                    .map(|x| x.into_bigint())
746                    .collect::<Vec<_>>(),
747            )
748            .into_affine();
749
750            let r = G::Group::msm_bigint(
751                &[g_hi, &[self.h, u_base]].concat(),
752                &[a_lo, &[rand_r, inner_prod(a_lo, b_hi)]]
753                    .concat()
754                    .iter()
755                    .map(|x| x.into_bigint())
756                    .collect::<Vec<_>>(),
757            )
758            .into_affine();
759
760            lr.push((l, r));
761            blinders.push((rand_l, rand_r));
762
763            sponge.absorb_g(&[l]);
764            sponge.absorb_g(&[r]);
765
766            // Round #i challenges;
767            // - not to be confused with "u_base"
768            // - not to be confused with "u" as "polyscale"
769            let u_pre = squeeze_prechallenge(&mut sponge);
770            let u = u_pre.to_field(&endo_r);
771            let u_inv = u.inverse().unwrap();
772
773            chals.push(u);
774            chal_invs.push(u_inv);
775
776            // IPA-folding polynomial coefficients
777            a = a_hi
778                .par_iter()
779                .zip(a_lo)
780                .map(|(&hi, &lo)| {
781                    // lo + u_inv * hi
782                    let mut res = hi;
783                    res *= u_inv;
784                    res += &lo;
785                    res
786                })
787                .collect();
788
789            // IPA-folding evaluation points
790            b = b_lo
791                .par_iter()
792                .zip(b_hi)
793                .map(|(&lo, &hi)| {
794                    // lo + u * hi
795                    let mut res = hi;
796                    res *= u;
797                    res += &lo;
798                    res
799                })
800                .collect();
801
802            // IPA-folding bases
803            g = G::combine_one_endo(endo_r, endo_q, g_lo, g_hi, u_pre);
804        }
805
806        assert!(
807            g.len() == 1 && a.len() == 1 && b.len() == 1,
808            "IPA commitment folding must produce single elements after log rounds"
809        );
810        let a0 = a[0];
811        let b0 = b[0];
812        let g0 = g[0];
813
814        // Compute r_prime, a folded blinder. It combines blinders on
815        // each individual step of the IPA folding process together
816        // with the final blinding_factor of the polynomial.
817        //
818        // r_prime := ∑_i (rand_l[i] * u[i]^{-1} + rand_r * u[i])
819        //          + blinding_factor
820        //
821        // where u is a vector of folding challenges, and rand_l/rand_r are
822        // intermediate L/R blinders.
823        let r_prime = blinders
824            .iter()
825            .zip(chals.iter().zip(chal_invs.iter()))
826            .map(|((rand_l, rand_r), (u, u_inv))| ((*rand_l) * u_inv) + (*rand_r * u))
827            .fold(blinding_factor, |acc, x| acc + x);
828
829        let d = <G::ScalarField as UniformRand>::rand(rng);
830        let r_delta = <G::ScalarField as UniformRand>::rand(rng);
831
832        // Compute delta, the commitment
833        // delta = [d] G0 + \
834        //         [b0*d] U_base + \
835        //         [r_delta] H^r (as a group element, in additive notation)
836        let delta = ((g0.into_group() + (u_base.mul(b0))).into_affine().mul(d)
837            + self.h.mul(r_delta))
838        .into_affine();
839
840        sponge.absorb_g(&[delta]);
841        let c = ScalarChallenge::new(sponge.challenge()).to_field(&endo_r);
842
843        // (?) Schnorr-like responses showing the knowledge of r_prime and a0.
844        let z1 = a0 * c + d;
845        let z2 = r_prime * c + r_delta;
846
847        OpeningProof {
848            delta,
849            lr,
850            z1,
851            z2,
852            sg: g0,
853        }
854    }

856    fn lagrange_basis(&self, domain: D<G::ScalarField>) -> Vec<PolyComm<G>> {
857        let n = domain.size();
858
859        // Let V be a vector space over the field F.
860        //
861        // Given
862        // - a domain [ 1, w, w^2, ..., w^{n - 1} ]
863        // - a vector v := [ v_0, ..., v_{n - 1} ] in V^n
864        //
865        // the FFT algorithm computes the matrix application
866        //
867        // u = M(w) * v
868        //
869        // where
870        // M(w) =
871        //   1 1       1           ... 1
872        //   1 w       w^2         ... w^{n-1}
873        //   ...
874        //   1 w^{n-1} (w^2)^{n-1} ... (w^{n-1})^{n-1}
875        //
876        // The IFFT algorithm computes
877        //
878        // v = M(w)^{-1} * u
879        //
880        // Let's see how we can use this algorithm to compute the lagrange basis
881        // commitments.
882        //
883        // Let V be the vector space F[x] of polynomials in x over F.
884        // Let v in V be the vector [ L_0, ..., L_{n - 1} ] where L_i is the i^{th}
885        // normalized Lagrange polynomial (where L_i(w^j) = j == i ? 1 : 0).
886        //
887        // Consider the rows of M(w) * v. Let me write out the matrix and vector
888        // so you can see more easily.
889        //
890        //   | 1 1       1           ... 1               |   | L_0     |
891        //   | 1 w       w^2         ... w^{n-1}         | * | L_1     |
892        //   | ...                                       |   | ...     |
893        //   | 1 w^{n-1} (w^2)^{n-1} ... (w^{n-1})^{n-1} |   | L_{n-1} |
894        //
895        // The 0th row is L_0 + L1 + ... + L_{n - 1}. So, it's the polynomial
896        // that has the value 1 on every element of the domain.
897        // In other words, it's the polynomial 1.
898        //
899        // The 1st row is L_0 + w L_1 + ... + w^{n - 1} L_{n - 1}. So, it's the
900        // polynomial which has value w^i on w^i.
901        // In other words, it's the polynomial x.
902        //
903        // In general, you can see that row i is in fact the polynomial x^i.
904        //
905        // Thus, M(w) * v is the vector u, where u = [ 1, x, x^2, ..., x^n ]
906        //
907        // Therefore, the IFFT algorithm, when applied to the vector u (the
908        // standard monomial basis) will yield the vector v of the (normalized)
909        // Lagrange polynomials.
910        //
911        // Now, because the polynomial commitment scheme is additively
912        // homomorphic, and because the commitment to the polynomial x^i is just
913        // self.g[i], we can obtain commitments to the normalized Lagrange
914        // polynomials by applying IFFT to the vector self.g[0..n].
915        //
916        //
917        // Further still, we can do the same trick for 'chunked' polynomials.
918        //
919        // Recall that a chunked polynomial is some f of degree k*n - 1 with
920        // f(x) = f_0(x) + x^n f_1(x) + ... + x^{(k-1) n} f_{k-1}(x)
921        // where each f_i has degree n-1.
922        //
923        // In the above, if we set u = [ 1, x^2, ... x^{n-1}, 0, 0, .., 0 ]
924        // then we effectively 'zero out' any polynomial terms higher than
925        // x^{n-1}, leaving us with the 'partial Lagrange polynomials' that
926        // contribute to f_0.
927        //
928        // Similarly, u = [ 0, 0, ..., 0, 1, x^2, ..., x^{n-1}, 0, 0, ..., 0]
929        // with n leading zeros 'zeroes out' all terms except the 'partial
930        // Lagrange polynomials' that contribute to f_1, and likewise for each
931        // f_i.
932        //
933        // By computing each of these, and recollecting the terms as a vector of
934        // polynomial commitments, we obtain a chunked commitment to the L_i
935        // polynomials.
936        let srs_size = self.g.len();
937        let num_elems = n.div_ceil(srs_size);
938        let mut chunks = Vec::with_capacity(num_elems);
939
940        // For each chunk
941        for i in 0..num_elems {
942            // Initialize the vector with zero curve points
943            let mut lg: Vec<<G as AffineRepr>::Group> = vec![<G as AffineRepr>::Group::zero(); n];
944            // Overwrite the terms corresponding to that chunk with the SRS
945            // curve points
946            let start_offset = i * srs_size;
947            let num_terms = min((i + 1) * srs_size, n) - start_offset;
948            for j in 0..num_terms {
949                lg[start_offset + j] = self.g[j].into_group()
950            }
951            // Apply the IFFT
952            domain.ifft_in_place(&mut lg);
953            // Append the 'partial Langrange polynomials' to the vector of elems
954            // chunks
955            chunks.push(<G as AffineRepr>::Group::normalize_batch(lg.as_mut_slice()));
956        }
957
958        (0..n)
959            .map(|i| PolyComm {
960                chunks: chunks.iter().map(|v| v[i]).collect(),
961            })
962            .collect()
963    }
964}

1025/// Commitment round challenges (endo mapped) and their inverses.

1027    pub chal: Vec<F>,
1028    pub chal_inv: Vec<F>,
1029}

1031impl<G: AffineRepr, const FULL_ROUNDS: usize> OpeningProof<G, FULL_ROUNDS> {
1032    /// Computes a log-sized vector of scalar challenges for
1033    /// recombining elements inside the IPA.
1034    pub fn prechallenges<EFqSponge: FqSponge<G::BaseField, G, G::ScalarField, FULL_ROUNDS>>(
1035        &self,
1036        sponge: &mut EFqSponge,
1037    ) -> Vec<ScalarChallenge<G::ScalarField>> {
1038        let _t = sponge.challenge_fq();
1039        self.lr
1040            .iter()
1041            .map(|(l, r)| {
1042                sponge.absorb_g(&[*l]);
1043                sponge.absorb_g(&[*r]);
1044                squeeze_prechallenge(sponge)
1045            })
1046            .collect()
1047    }
1048
1049    /// Same as `prechallenges`, but maps scalar challenges using the provided
1050    /// endomorphism, and computes their inverses.
1051    pub fn challenges<EFqSponge: FqSponge<G::BaseField, G, G::ScalarField, FULL_ROUNDS>>(
1052        &self,
1053        endo_r: &G::ScalarField,
1054        sponge: &mut EFqSponge,
1055    ) -> Challenges<G::ScalarField> {
1056        let chal: Vec<_> = self
1057            .lr
1058            .iter()
1059            .map(|(l, r)| {
1060                sponge.absorb_g(&[*l]);
1061                sponge.absorb_g(&[*r]);
1062                squeeze_challenge(endo_r, sponge)
1063            })
1064            .collect();
1065
1066        let chal_inv = {
1067            let mut cs = chal.clone();
1068            ark_ff::batch_inversion(&mut cs);
1069            cs
1070        };
1071
1072        Challenges { chal, chal_inv }
1073    }
