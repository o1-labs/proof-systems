7/// Represent a polynomial either with its coefficients or its evaluations
8pub enum DensePolynomialOrEvaluations<'a, F: FftField, D: EvaluationDomain<F>> {
9    /// Polynomial represented by its coefficients
10    DensePolynomial(&'a DensePolynomial<F>),
11    /// Polynomial represented by its evaluations over a domain D
12    Evaluations(&'a Evaluations<F, D>, D),
13}
14
15/// A formal sum of the form
16/// `s_0 * p_0 + ... s_n * p_n`
17/// where each `s_i` is a scalar and each `p_i` is a polynomial.
18/// The parameter `P` is expected to be the coefficients of the polynomial
19/// `p_i`, even though we could treat it as the evaluations.
20///
21/// This hypothesis is important if `to_dense_polynomial` is called.
22#[derive(Default)]
23struct ScaledChunkedPolynomial<F, P>(Vec<(F, P)>);
24
25impl<F, P> ScaledChunkedPolynomial<F, P> {
26    fn add_poly(&mut self, scale: F, p: P) {
27        self.0.push((scale, p))
28    }
29}
30
31impl<'a, F: Field> ScaledChunkedPolynomial<F, &'a [F]> {
32    /// Compute the resulting scaled polynomial.
33    /// Example:
34    /// Given the two polynomials `1 + 2X` and `3 + 4X`, and the scaling
35    /// factors `2` and `3`, the result is the polynomial `11 + 16X`.
36    /// ```text
37    /// 2 * [1, 2] + 3 * [3, 4] = [2, 4] + [9, 12] = [11, 16]
38    /// ```
39    fn to_dense_polynomial(&self) -> DensePolynomial<F> {
40        // Note: using a reference to avoid reallocation of the result.
41        let mut res = DensePolynomial::<F>::zero();
42
43        let scaled: Vec<_> = self
44            .0
45            .par_iter()
46            .map(|(scale, segment)| {
47                let scale = *scale;
48                // We simply scale each coefficients.
49                // It is simply because DensePolynomial doesn't have a method
50                // `scale`.
51                let v = segment.par_iter().map(|x| scale * *x).collect();
52                DensePolynomial::from_coefficients_vec(v)
53            })
54            .collect();
55
56        for p in scaled {
57            res += &p;
58        }
59
60        res
61    }
62}

64/// Combine the polynomials using a scalar (`polyscale`), creating a single
65/// unified polynomial to open. This function also accepts polynomials in
66/// evaluations form. In this case it applies an IFFT, and, if necessary,
67/// applies chunking to it (ie. split it in multiple polynomials of
68/// degree less than the SRS size).
69///
70/// Parameters:
71/// - `plnms`: vector of polynomials, either in evaluations or coefficients form, together with
72///    a set of scalars representing their blinders.
73/// - `polyscale`: scalar to combine the polynomials, which will be scaled based on the number of
74///    polynomials to combine.
75///
76/// Output:
77/// - `combined_poly`: combined polynomial. The order of the output follows the order of `plnms`.
78/// - `combined_comm`: combined scalars representing the blinder commitment.
79///
80/// Example:
81/// Given the three polynomials `p1(X)`, and `p3(X)` in coefficients
82/// forms, p2(X) in evaluation form,
83/// and the scaling factor `polyscale`, the result will be the polynomial:
84///
85/// ```text
86/// p1(X) + polyscale * i_fft(chunks(p2))(X) + polyscale^2 p3(X)
87/// ```
88///
89/// Additional complexity is added to handle chunks.
90pub fn combine_polys<G: CommitmentCurve, D: EvaluationDomain<G::ScalarField>>(
91    plnms: PolynomialsToCombine<G, D>,
92    polyscale: G::ScalarField,
93    srs_length: usize,
94) -> (DensePolynomial<G::ScalarField>, G::ScalarField) {
95    // Initialising the output for the combined coefficients forms
96    let mut plnm_coefficients =
97        ScaledChunkedPolynomial::<G::ScalarField, &[G::ScalarField]>::default();
98    // Initialising the output for the combined evaluations forms
99    let mut plnm_evals_part = {
100        // For now just check that all the evaluation polynomials are the same
101        // degree so that we can do just a single FFT.
102        // If/when we change this, we can add more complicated code to handle
103        // different degrees.
104        let degree = plnms
105            .iter()
106            .fold(None, |acc, (p, _)| match p {
107                DensePolynomialOrEvaluations::DensePolynomial(_) => acc,
108                DensePolynomialOrEvaluations::Evaluations(_, d) => {
109                    if let Some(n) = acc {
110                        assert_eq!(n, d.size());
111                    }
112                    Some(d.size())
113                }
114            })
115            .unwrap_or(0);
116        vec![G::ScalarField::zero(); degree]
117    };
118
119    // Will contain ∑ comm_chunk * polyscale^i
120    let mut combined_comm = G::ScalarField::zero();
121
122    // Will contain polyscale^i
123    let mut polyscale_to_i = G::ScalarField::one();
124
125    // Iterating over polynomials in the batch.
126    // Note that the chunks in the commitment `p_i_comm` are given as `PolyComm<G::ScalarField>`. They are
127    // evaluations.
128    // We do modify two different structures depending on the form of the
129    // polynomial we are currently processing: `plnm` and `plnm_evals_part`.
130    // We do need to treat both forms separately.
131    for (p_i, p_i_comm) in plnms {
132        match p_i {
133            // Here we scale the polynomial in evaluations forms
134            // Note that based on the check above, sub_domain.size() always give
135            // the same value
136            DensePolynomialOrEvaluations::Evaluations(evals_i, sub_domain) => {
137                let stride = evals_i.evals.len() / sub_domain.size();
138                let evals = &evals_i.evals;
139                plnm_evals_part
140                    .par_iter_mut()
141                    .enumerate()
142                    .for_each(|(i, x)| {
143                        *x += polyscale_to_i * evals[i * stride];
144                    });
145                for comm_chunk in p_i_comm.into_iter() {
146                    combined_comm += &(*comm_chunk * polyscale_to_i);
147                    polyscale_to_i *= &polyscale;
148                }
149            }
150
151            // Here we scale the polynomial in coefficient forms
152            DensePolynomialOrEvaluations::DensePolynomial(p_i) => {
153                let mut offset = 0;
154                // iterating over chunks of the polynomial
155                for comm_chunk in p_i_comm.into_iter() {
156                    let segment = &p_i.coeffs[std::cmp::min(offset, p_i.coeffs.len())
157                        ..std::cmp::min(offset + srs_length, p_i.coeffs.len())];
158                    plnm_coefficients.add_poly(polyscale_to_i, segment);
159
160                    combined_comm += &(*comm_chunk * polyscale_to_i);
161                    polyscale_to_i *= &polyscale;
162                    offset += srs_length;
163                }
164            }
165        }
166    }
167
168    // Now, we will combine both evaluations and coefficients forms
169
170    // plnm will be our final combined polynomial. We first treat the
171    // polynomials in coefficients forms, which is simply scaling the
172    // coefficients and add them.
173    let mut combined_plnm = plnm_coefficients.to_dense_polynomial();
174
175    if !plnm_evals_part.is_empty() {
176        // n is the number of evaluations, which is a multiple of the
177        // domain size.
178        // We treat now each chunk.
179        let n = plnm_evals_part.len();
180        let max_poly_size = srs_length;
181        // equiv to divceil, but unstable in rust < 1.73.
182        let num_chunks = n / max_poly_size + if n % max_poly_size == 0 { 0 } else { 1 };
183        // Interpolation on the whole domain, i.e. it can be d2, d4, etc.
184        combined_plnm += &Evaluations::from_vec_and_domain(plnm_evals_part, D::new(n).unwrap())
185            .interpolate()
186            .to_chunked_polynomial(num_chunks, max_poly_size)
187            .linearize(polyscale);
188    }
189
190    (combined_plnm, combined_comm)
191}
