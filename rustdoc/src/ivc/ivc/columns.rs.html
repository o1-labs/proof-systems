1//! IVC circuit layout - Top level documentation outdated
2//!
3//! The IVC circuit is tiled vertically. We assume we have as many
4//! rows as we need: if we don't, we wrap around and continue.
5//!
6//! The biggest blocks are hashes and ECAdds, so other blocks may be wider.
7//!
8//! `N := N_IVC + N_APP` is the total number of columns in the circuit.
9//!
10//! Vertically stacked blocks are as follows:
11//!
12//!```text
13//!
14//!         Inputs:
15//!      Each point is 2 base field coordinates in 17 15-bit limbs
16//!       recomposed as 8 75-bit limbs
17//!       recomposed as 4 150-bit limbs.
18//!
19//!              34            8      4
20//!            Input1         R75   R150
21//!  1   |-----------------|-------|----|
22//!      |      C_L1       |       |    |
23//!      |      C_L2       |       |    |
24//!      |                 |       |    |
25//!      |      ...        |       |    |
26//!   N  |-----------------|       |    |
27//!      |      C_R1       |       |    |
28//!      |                 |       |    |
29//!      |      ...        |       |    |
30//!  2N  |-----------------|       |    |
31//!      |      C_O1       |       |    |
32//!      |                 |       |    |
33//!      |      ...        |       |    |
34//!  3N  |-----------------|-------|----|
35//!      0       ...     34*2    76    80
36//!
37//!
38//!                      Hashes
39//!     (one hash at a row, passing data to the next one)
40//!     (for i∈N, the input row #i containing 4 150-bit elements
41//!      is processed by hash rows 2*i and 2*i+1)
42//!
43//!  1   |------------------------------------------|
44//!      |                                          |
45//!      |                                         .| . here is h_l
46//!  2N  |------------------------------------------|   must be equal to public input!
47//!      |                                          |                    (H_i in nova)
48//!      |                                         .| . here is h_r
49//!  4N  |------------------------------------------|
50//!      |                                          |
51//!      |                                         .| . here is h_o, equal to H_{i+1} in Nova
52//!  6N  |------------------------------------------|
53//!      |                                         .| r = h_lr = h(h_l,h_r)
54//!      |                                         .| ϕ = h_lro = h(r,h_o)
55//! 6N+2 |------------------------------------------|
56//!
57//!       TODO: we also need to squeeze challenges for
58//!       the right (strict) instance: β, γ, j (joint_combiner)
59//!
60//!       TODO: we can hash (x0,x1+b*2^150) instead of (x0,x1,y0,y1).
61//!
62//! Scalars block.
63//!
64//! Most of the r^2 and r^3 cells are /unused/, and this design can be
65//! much more optimal if r^2|r^3 elements come in a separate block.
66//! But the overhead is not big and it's a very easy layout, so
67//! keeping it for now.
68//!
69//!     constϕ
70//!         constr
71//!                ϕ^i             r^3·ϕ^i    ϕ^i        r*ϕ^i        r^2·ϕ^i_k    r^3·ϕ^i_k
72//!                      r*ϕ^i             in 17 limbs  in 17 limbs      ...         ...
73//!                           r^2·ϕ^i         each        each
74//!   1  |---|---|-----|-------|----|----|------------|------------|------------|------------|
75//!      | ϕ   r    ϕ     rϕ    r^2ϕ r^3ϕ|            |            |            |            |
76//!      | ϕ   r   ϕ^2   rϕ^2            |            |            |            |            |
77//!      | ϕ   r   ϕ^3   rϕ^3            |            |            |            |            |
78//!      |                               |            |            |            |            |
79//!      |                               |            |            |            |            |
80//!      |                               |            |            |            |            |
81//!  i   |                               |            |            |            |            |
82//!      |                               |            |            |            |            |
83//!      |                               |            |            |            |            |
84//!      |                               |            |            |            |            |
85//!      |       ϕ^{N+1}                 |            |            |            |            |
86//!  N+1 |-------------------------------|------------|------------|------------|------------|
87//!       1    2   3   4      5      6   ...        6+17          6+2*17                    6+4*17
88//!
89//!
90//! We compute the following equations, where equations in "quotes" are
91//! what we /want/ to prove, and non-quoted is an equavilant version
92//! that we actually prove instead:
93//! - "C_{O,i} = C_{L,i} + r·C_{R,i}":
94//!   - C_{O,i} = C_{L,i} + C_{R',i}
95//!   - "C_{R',i} = r·C_{R,i}"
96//!     - bucket[(ϕ^i)_k] -= C_{R',i}
97//!     - bucket[(r·ϕ^i)_k] += C_{R,i}
98//! - "E_O = E_L + r·T_0 + r^2·T_1 + r^3·E_R":
99//!   - E_O = E_L + E_R'
100//!   - "E_R' = r·T_0 + r^2·T_1 + r^3·E_R"
101//!     - bucket[(ϕ^{n+1})_k] += E_R'
102//!     - bucket[(r·ϕ^{n+1})_k] += T_0
103//!     - bucket[(r^2·ϕ^{n+1})_k] += T_1
104//!     - bucket[(r^3·ϕ^{n+1})_k] += E_R
105//!
106//! Runtime access time is represented by ? because it's not known in advance.
107//!
108//! Output and input RAM invocations in the same row use the same coeff/memory index.
109//!
110//! TODO FIXME we need to write into /different/ buckets.
111//!
112//! FEC Additions, one per row, each one is ~230 columns:
113//!
114//!                  .           input #2 (looked up)                  .                  Output
115//!                  .                  Access        input#2          .     FEC ADD      access    output
116//!          input#1 .  Coeff/mem ix     Time         Value            .   computation     time     value
117//!   1   |-------------------------------------------------------------------------------|-----|------------|
118//!       | C_{R',1} | ϕ^1_0           |  ?  | bucket[ϕ^1_0]           |                  |  ?  | newbucket  |
119//!       | C_{R',2} | ϕ^2_0           |  ?  | bucket[ϕ^2_0]           |                  |  ?  | newbucket  |
120//!       |          |      ...        | ... |                         |                  | ... |            |
121//!       | C_{R',N} | ϕ^N_0           |  ?  | bucket[ϕ^N_0]           |                  |  ?  | newbucket  |
122//!       | C_{R',1} | ϕ^1_1           |  ?  | bucket[ϕ^0_1]           |                  |  ?  | newbucket  |
123//!       |          |      ...        | ... |                         |                  | ... |            |
124//!       | C_{R',i} | ϕ^i_k           |  ?  | bucket[ϕ^i_k]           |                  |  ?  | newbucket  |
125//!       |          |                 |     |                         |                  |     |            |
126//!       |          |      ...        | ... |                         |                  | ... |            |
127//!  17*N |--------------------------------------------------------------------------------------------------|
128//!       | C_{R,i}  | r·(ϕ^i_k)       |  ?  | bucket[r·(ϕ^i_k)]       |                  |  ?  | newbucket  |
129//!       |          |                 |     |                         |                  |     |            |
130//!       |          |      ...        | ... |                         |                  | ... |            |
131//!  34*N |--------------------------------------------------------------------------------------------------|
132//!       |-C_{R',i} |      -          |  -  | C_{L,i}                 |                  |  -  | C_{O,i}    |
133//!       |          |     ...         | ... |                         |                  | ... |            |
134//!  35*N |--------------------------------------------------------------------------------------------------|
135//!       | -E_R'    | ϕ^{n+1}_k       |  ?  | bucket[ϕ^{n+1}_k]       |                  |     | newbucket  |
136//!       |          |     ...         | ... |                         |                  | ... |            |
137//!       |  T_0     | r·(ϕ^{n+1})     |  ?  | bucket[r·(ϕ^{n+1})]     |                  |     | newbucket  |
138//!       |          |     ...         | ... |                         |                  | ... |            |
139//!       |  T_1     | r^2·(ϕ^{n+1}_k) |  ?  | bucket[r^2·(ϕ^{n+1}_k)] |                  |     | newbucket  |
140//!       |          |     ...         | ... |                         |                  | ... |            |
141//!       |  E_R     | r^3·(ϕ^{n+1}_k) |  ?  | bucket[r^3·(ϕ^{n+1}_k)] |                  |     | newbucket  |
142//!       |          |     ...         | ... |                         |                  | ... |            |
143//! 35*N+ |  E_L     |      -          |  -  |      E_R'               |                  |  -  |    E_O     |
144//! 4*17+ |--------------------------------------------------------------------------------------------------|
145//! 1
146//!
147//! TODO: add different challenges: β, γ, joint_combiner
148//!
149//! Challenges block.
150//!
151//!            relaxed
152//!                       strict
153//!                    (relaxed in-place)
154//!        r   α_{L,i}    α_{R}^i     α_{O,i}
155//!  1    |--|--------|-----------|-----------------------|
156//!       |  |        | α_R = h_R |                       |
157//!       |  |        |           |                       |
158//!       |  |        |           |                       |
159//!       |  |        | α_R^i     | α_{L,i} + r·α_{R,i}^i |
160//!       |  |        |           |                       |
161//!       |  |        |           |                       |
162//!       |  |        |           |                       |
163//!       |  |        |           |                       |
164//!       |  |        |           |                       |
165//!       |  |        |           |                       |
166//! #chal |--|--------|-----------|-----------------------|
167//!
168//! #chal is the number of constraints. Our optimistic expectation is
169//! that it is around const*N for const < 3.
170//!
171//!
172//! "u" block. In the general form we want to prove
173//! u_O = u_L + r·u_R, but u_R = 0, so we prove
174//! u_O = u_L + r.
175//!
176//!     r    u_L       u_O = u_L + r
177//!    |--|--------|--------------------|
178//!    |--|--------|--------------------|
179//!
180//!
181//! 2^15 |---- --------------------------------------|
182//!```
183//!
184//!
185//! Assume that IVC circuit takess CELL cells, e.g. CELL = 10000*N.
186//! Then we can calculate N_IVC as dependency of N_APP in this way:
187//! ```text
188//!    N = N_APP + (CELL/2^15)*N
189//!    (1 - CELL/2^15)*N = N_APP
190//!    N = (1/(1 - CELL/2^15)) * N_APP = (2^15 / (2^15 - CELL)) * N_APP
191//!    N_IVC = (1/(1 - CELL/2^15) - 1) * N_APP = (2^15 / (2^15 - CELL) - 1) * N_APP
192//! ```
193//!
194//! --- (slightly) OUTDATED BELOW ---
195//!
196//! Counting cells:
197//! - Inputs:              2 * 17 * 3N = 102N
198//! - Inputs repacked 75:  2 * 4 * 3N = 24N
199//! - Inputs repacked 150: 2 * 2 * 3N = 12N
200//! - Hashes:              2 * 165 * 3N = 990N (max 4 * 165 * 3N if we add 165 constants to every call)
201//! - scalars:             4 N + 17 * 3 * N = 55 N
202//! - ECADDs:              230 * 35 * N = 8050N
203//!
204//! Total (CELL):         ~9233*N
205//!
206//! ...which is less than 32k*N
207//!
208//! In our particular case, CELL = 9233, so
209//!    N_IVC = 0.39 N_APP
210//!
211//! Which means for e.g. 50 columns we'll need extra 20 of IVC.

230/// Number of blocks in the circuit.
231pub const N_BLOCKS: usize = 6;
232
233/// Defines the height of each block in the IVC circuit.
234/// As described in the top-level of this file, the circuit is tiled
235/// vertically, and the height of each block is defined by the actual
236/// application that needs to be run.
237pub fn block_height<const N_COL_TOTAL: usize, const N_CHALS: usize>(block_num: usize) -> usize {
238    match block_num {
239        // The first block is used for the decomposition of the inputs.
240        // As we do have 3 inputs (left, right and output), we need 3 times the
241        // total number of columns.
242        0 => 3 * N_COL_TOTAL,
243        // The second block is used for hashing the different values.
244        // We do have 2 foreign field elements to hash per commitment.
245        // At the moment, we do split each foreign field element in 150 bits
246        // chunk, therefore we need to hash (2 * 2 * 3) (= 12) scalar field
247        // elements. As our Poseidon state can handle 2 elements per absorb, we
248        // need to compute hashes.
249        // FIXME: why an additional 2?
250        // FIXME: use the encoding of the bit of y instead. It would reduce to 3
251        // hashes instead of 6.
252        //   For a EC point (x, y), we can only rely on x and the bit sign of y.
253        //   We decompose x into `x_{0..149}` and `x_{150..255}` (only 255 as
254        //   the foreign field is 255 bits), and we add into the value
255        //   `x_{150..255}` the bit sign of `y`. From there, we can hash the two
256        //   values, which fits in our state of size 3.
257        1 => 6 * N_COL_TOTAL + 2,
258        // The third block is used for the randomisation of the MSM.
259        2 => N_COL_TOTAL + 1,
260        // The fourth block is used for the foreign field ECC addition.
261        3 => 35 * N_COL_TOTAL + 5,
262        // The fifth block is used for the challenges.
263        // The total number of challenges is given as a type parameter.
264        4 => N_CHALS,
265        // The sixth block is used for the homogeneised value `u`.
266        // We do only need one row for this one as there is a single value to
267        // randomize
268        5 => 1,
269        _ => panic!("block_size: no block number {block_num:?}"),
270    }
271}
272
273/// Returns total height of the IVC circuit.
274pub fn total_height<const N_COL_TOTAL: usize, const N_CHALS: usize>() -> usize {
275    (0..N_BLOCKS)
276        .map(block_height::<N_COL_TOTAL, N_CHALS>)
277        .sum()
278}

286pub enum IVCColumn {
287    /// A single column containing the folding iteration number,
288    /// starting with 0 for the base case, and non-zero positive for
289    /// inductive case.
290    FoldIteration,
291
292    /// Selector for blocks. Inner usize is ∈ [0,#blocks).
293    /// The selectors are fixed values for each instance of the relation.
294    /// The different blocks are described below or at the top-level of this
295    /// file.
296    BlockSel(usize),
297
298    /// 2*17 15-bit limbs (two base field points)
299    Block1Input(usize),
300    /// 2*4 75-bit limbs
301    Block1InputRepacked75(usize),
302    /// 2*2 150-bit limbs
303    Block1InputRepacked150(usize),
304
305    /// 1 hash per row
306    // FIXME/TODO: we might want to have 2 additional columns for the actual
307    // input we would like to hash, and must constrain the input state of the
308    // hash to be the sum of the previous state.
309    // i.e. for each hash, we have 2 + 3 + 2 columns
310    // - x, y the inputs we want to absorb
311    // - s0, s1, s2 the initial state of the hash
312    // - s0 + x, s1 + y the final state of the hash, with s2 being unused.
313    // - s0 + x, s1 + y and s2 are considered as the column
314    // PoseidonColumn::Input of the Poseidon gadget, and a constrain must be
315    // added between s0 and s0 + x (resp. s1 and s1 + y).
316    Block2Hash(
317        PoseidonColumn<
318            IVC_POSEIDON_STATE_SIZE,
319            IVC_POSEIDON_NB_FULL_ROUND,
320            IVC_POSEIDON_NB_PARTIAL_ROUND,
321        >,
322    ),
323
324    /// Constant phi
325    Block3ConstPhi,
326    /// Constant r
327    Block3ConstR,
328    /// Scalar coeff #1, powers of Phi, phi^i
329    Block3PhiPow,
330    /// Scalar coeff #2, r * phi^i
331    Block3PhiPowR,
332    /// Scalar coeff #2, r^2 * phi^i
333    Block3PhiPowR2,
334    /// Scalar coeff #2, r^3 * phi^i
335    Block3PhiPowR3,
336    /// 17 15-bit limbs
337    Block3PhiPowLimbs(usize),
338    /// 17 15-bit limbs
339    Block3PhiPowRLimbs(usize),
340    /// 17 15-bit limbs
341    Block3PhiPowR2Limbs(usize),
342    /// 17 15-bit limbs
343    Block3PhiPowR3Limbs(usize),
344
345    /// 2*4 75-bit limbs
346    Block4Input1(usize),
347    /// Coeffifient which is indexing a bucket. Used for both lookups in this row.
348    Block4Coeff,
349    /// RAM lookup access time for input 1.
350    Block4Input2AccessTime,
351    /// 2*4 75-bit limbs
352    Block4Input2(usize),
353    /// EC ADD intermediate wires
354    Block4ECAddInter(FECColumnInter),
355    /// 2*17 15-bit limbs
356    Block4OutputRaw(FECColumnOutput),
357    // TODO: this might be just Input2AccessTime + 1?
358    /// RAM lookup access time for output.
359    Block4OutputAccessTime,
360    /// 2*4 75-bit limbs
361    Block4OutputRepacked(usize),
362
363    /// Constant h_r
364    Block5ConstHr,
365    /// Constant r
366    Block5ConstR,
367    /// α_{L,i}
368    Block5ChalLeft,
369    /// α_R^i, where α_R = h_R
370    Block5ChalRight,
371    /// α_{O,i} = α_{L,i} + r·α_R^i
372    Block5ChalOutput,
373
374    /// Constant r
375    Block6ConstR,
376    /// u_L
377    Block6ULeft,
378    /// u_O = u_L + r
379    Block6UOutput,

382impl ColumnIndexer<usize> for IVCColumn {
383    /// Number of columns used by the IVC circuit
384    /// It contains at least the columns used for Poseidon.
385    /// It does not include the additional columns that might be required
386    /// to reduce to degree 2.
387    /// FIXME: This can be improved by changing a bit the layer.
388    /// The reduction to degree 2 should happen in the gadgets to avoid adding
389    /// extra columns and leave sparse rows.
390    // We consider IVCPoseidonColumn::N_COL but it should be the maximum of
391    // the different gadgets/blocks.
392    // We also add 1 for the FoldIteration column.
393    const N_COL: usize = IVCPoseidonColumn::N_COL + 1 + N_BLOCKS;
394
395    fn to_column(self) -> Column<usize> {
396        match self {
397            // We keep a column that will be used for the folding iteration.
398            // Question: do we need it for all the rows or does it appear only
399            // for one hash?
400            IVCColumn::FoldIteration => Column::Relation(0),
401            IVCColumn::BlockSel(i) => {
402                assert!(i < N_BLOCKS);
403                Column::FixedSelector(i)
404            }
405
406            IVCColumn::Block1Input(i) => {
407                assert!(i < 2 * N_LIMBS_SMALL);
408                Column::Relation(i).add_rel_offset(1)
409            }
410            IVCColumn::Block1InputRepacked75(i) => {
411                assert!(i < 2 * N_LIMBS_LARGE);
412                Column::Relation(2 * N_LIMBS_SMALL + i).add_rel_offset(1)
413            }
414            IVCColumn::Block1InputRepacked150(i) => {
415                assert!(i < 2 * N_LIMBS_XLARGE);
416                Column::Relation(2 * N_LIMBS_SMALL + 2 * N_LIMBS_LARGE + i).add_rel_offset(1)
417            }
418
419            // The second block of columns will be used for hashing the
420            // different values, like the commitments and the challenges.
421            IVCColumn::Block2Hash(poseidon_col) => match poseidon_col {
422                // We map the round constants into the appropriate fixed
423                // selector.
424                // We suppose that the round constants are added after the
425                // public values describing the block selection.
426                PoseidonColumn::RoundConstant(round, state_size) => {
427                    let idx = round * IVC_POSEIDON_STATE_SIZE + state_size;
428                    Column::FixedSelector(idx + N_BLOCKS)
429                }
430                // We add a padding for the fold iteration.
431                // Even though, we might not need it.
432                other => other.to_column().add_rel_offset(1),
433            },
434
435            // The third block is used to compute the randomisation of the MSM
436            IVCColumn::Block3ConstPhi => Column::Relation(0).add_rel_offset(1),
437            IVCColumn::Block3ConstR => Column::Relation(1).add_rel_offset(1),
438            IVCColumn::Block3PhiPow => Column::Relation(2).add_rel_offset(1),
439            IVCColumn::Block3PhiPowR => Column::Relation(3).add_rel_offset(1),
440            IVCColumn::Block3PhiPowR2 => Column::Relation(4).add_rel_offset(1),
441            IVCColumn::Block3PhiPowR3 => Column::Relation(5).add_rel_offset(1),
442            IVCColumn::Block3PhiPowLimbs(i) => {
443                assert!(i < N_LIMBS_SMALL);
444                Column::Relation(6 + i).add_rel_offset(1)
445            }
446            IVCColumn::Block3PhiPowRLimbs(i) => {
447                assert!(i < N_LIMBS_SMALL);
448                Column::Relation(6 + N_LIMBS_SMALL + i).add_rel_offset(1)
449            }
450            IVCColumn::Block3PhiPowR2Limbs(i) => {
451                assert!(i < N_LIMBS_SMALL);
452                Column::Relation(6 + 2 * N_LIMBS_SMALL + i).add_rel_offset(1)
453            }
454            IVCColumn::Block3PhiPowR3Limbs(i) => {
455                assert!(i < N_LIMBS_SMALL);
456                Column::Relation(6 + 3 * N_LIMBS_SMALL + i).add_rel_offset(1)
457            }
458
459            // The fourth block is used for the foreign field ECC addition
460            IVCColumn::Block4Input1(i) => {
461                assert!(i < 2 * N_LIMBS_LARGE);
462                Column::Relation(i).add_rel_offset(1)
463            }
464            IVCColumn::Block4Coeff => Column::Relation(8).add_rel_offset(1),
465            IVCColumn::Block4Input2AccessTime => Column::Relation(9).add_rel_offset(1),
466            IVCColumn::Block4Input2(i) => {
467                assert!(i < 2 * N_LIMBS_LARGE);
468                Column::Relation(10 + i).add_rel_offset(1)
469            }
470            IVCColumn::Block4ECAddInter(fec_inter) => {
471                fec_inter.to_column().add_rel_offset(18).add_rel_offset(1)
472            }
473            IVCColumn::Block4OutputRaw(fec_output) => fec_output
474                .to_column()
475                .add_rel_offset(18 + FECColumnInter::N_COL)
476                .add_rel_offset(1),
477            IVCColumn::Block4OutputAccessTime => {
478                Column::Relation(18 + FECColumnInter::N_COL + FECColumnOutput::N_COL)
479                    .add_rel_offset(1)
480            }
481            IVCColumn::Block4OutputRepacked(i) => {
482                assert!(i < 2 * N_LIMBS_LARGE);
483                Column::Relation(18 + FECColumnInter::N_COL + FECColumnOutput::N_COL + 1 + i)
484                    .add_rel_offset(1)
485            }
486
487            // The fifth block is used for the challenges. Note that the
488            // challenges contains all the alphas (used to bundle the
489            // constraints) and also the challenges used by the PIOP.
490            IVCColumn::Block5ConstHr => Column::Relation(0).add_rel_offset(1),
491            IVCColumn::Block5ConstR => Column::Relation(1).add_rel_offset(1),
492            IVCColumn::Block5ChalLeft => Column::Relation(2).add_rel_offset(1),
493            IVCColumn::Block5ChalRight => Column::Relation(3).add_rel_offset(1),
494            IVCColumn::Block5ChalOutput => Column::Relation(4).add_rel_offset(1),
495
496            // The sixth block is used for the homogeneised value `u`.
497            IVCColumn::Block6ConstR => Column::Relation(0).add_rel_offset(1),
498            IVCColumn::Block6ULeft => Column::Relation(1).add_rel_offset(1),
499            IVCColumn::Block6UOutput => Column::Relation(2).add_rel_offset(1),
500        }
501    }
502}

504/// Lens used to convert between IVC columns and Poseidon columns.
505/// We use one row for each hash, whatever the type of inputs the hash has.
506pub struct IVCHashLens {}
507
508impl MPrism for IVCHashLens {
509    type Source = IVCColumn;
510    type Target = IVCPoseidonColumn;
511
512    // [traverse] must map the columns IVCColumn to the corresponding columns of
513    // the Poseidon circuit.
514    // We do suppose that the columns are ordered in the same way in the IVC and
515    // the Poseidon circuit, and that a whole row is used by the Poseidon gadget
516    fn traverse(&self, source: Self::Source) -> Option<Self::Target> {
517        match source {
518            IVCColumn::Block2Hash(col) => Some(col),
519            _ => None,
520        }
521    }
522
523    // [re_get] must map back the columns of the Poseidon gadget to the columns
524    // of the IVC circuit.
525    // As said for [traverse], a whole row, Block2Hash, is used by the Poseidon
526    // gadget.
527    // We map the Poseidon columns in the same order than described in
528    // [crate::poseidon_8_56_5_3_2::columns::PoseidonColumn], i.e. Input,
529    // Partialround and FullRound.
530    fn re_get(&self, target: Self::Target) -> Self::Source {
531        IVCColumn::Block2Hash(target)
532    }
533}

535pub struct IVCFECLens {}
536
537impl MPrism for IVCFECLens {
538    type Source = IVCColumn;
539    type Target = FECColumn;
540
541    fn traverse(&self, source: Self::Source) -> Option<Self::Target> {
542        match source {
543            IVCColumn::Block4Input1(i) => {
544                if i < 4 {
545                    Some(FECColumn::Input(FECColumnInput::XP(i)))
546                } else {
547                    Some(FECColumn::Input(FECColumnInput::YP(i)))
548                }
549            }
550            IVCColumn::Block4Input2(i) => {
551                if i < 4 {
552                    Some(FECColumn::Input(FECColumnInput::XQ(i)))
553                } else {
554                    Some(FECColumn::Input(FECColumnInput::YQ(i)))
555                }
556            }
557            IVCColumn::Block4OutputRaw(output) => Some(FECColumn::Output(output)),
558            IVCColumn::Block4ECAddInter(inter) => Some(FECColumn::Inter(inter)),
559            _ => None,
560        }
561    }
562
563    fn re_get(&self, target: Self::Target) -> Self::Source {
564        match target {
565            FECColumn::Input(FECColumnInput::XP(i)) => IVCColumn::Block4Input1(i),
566            FECColumn::Input(FECColumnInput::YP(i)) => IVCColumn::Block4Input1(4 + i),
567            FECColumn::Input(FECColumnInput::XQ(i)) => IVCColumn::Block4Input2(i),
568            FECColumn::Input(FECColumnInput::YQ(i)) => IVCColumn::Block4Input2(4 + i),
569            FECColumn::Output(output) => IVCColumn::Block4OutputRaw(output),
570            FECColumn::Inter(inter) => IVCColumn::Block4ECAddInter(inter),
571        }
572    }
573}
