84    /// Generic IVC circuit builder.
85    fn build_ivc_circuit<
86        RNG: RngCore + CryptoRng,
87        LT: LookupTableID,
88        L: MPrism<Source = LT, Target = IVCLookupTable<Ff1>>,
89    >(
90        rng: &mut RNG,
91        domain_size: usize,
92        fold_iteration: usize,
93        lt_lens: L,
94    ) -> IVCWitnessBuilderEnvRaw<LT> {
95        let mut witness_env = IVCWitnessBuilderEnvRaw::<LT>::create();
96
97        let mut comms_left: Box<_> = box_array![(Ff1::zero(),Ff1::zero()); TEST_N_COL_TOTAL];
98        let mut comms_right: Box<_> = box_array![(Ff1::zero(),Ff1::zero()); TEST_N_COL_TOTAL];
99        let mut comms_output: Box<_> = box_array![(Ff1::zero(),Ff1::zero()); TEST_N_COL_TOTAL];
100
101        for i in 0..TEST_N_COL_TOTAL {
102            comms_left[i] = (
103                <Ff1 as UniformRand>::rand(rng),
104                <Ff1 as UniformRand>::rand(rng),
105            );
106            comms_right[i] = (
107                <Ff1 as UniformRand>::rand(rng),
108                <Ff1 as UniformRand>::rand(rng),
109            );
110            comms_output[i] = (
111                <Ff1 as UniformRand>::rand(rng),
112                <Ff1 as UniformRand>::rand(rng),
113            );
114        }
115
116        println!("Building fixed selectors");
117
118        let fixed_selectors: [Vec<Fp>; N_FSEL_IVC] =
119            build_fixed_selectors::<TEST_N_COL_TOTAL, TEST_N_CHALS>(domain_size);
120
121        witness_env.set_fixed_selectors(fixed_selectors.to_vec());
122
123        println!("Calling the IVC circuit");
124        // TODO add nonzero E/T values.
125        ivc_circuit::<_, _, _, _, TEST_N_COL_TOTAL, TEST_N_CHALS>(
126            &mut SubEnvLookup::new(&mut witness_env, lt_lens),
127            fold_iteration,
128            comms_left,
129            comms_right,
130            comms_output,
131            [(Ff1::zero(), Ff1::zero()); 3],
132            [(Ff1::zero(), Ff1::zero()); 2],
133            Fp::zero(),
134            Box::new(
135                (*vec![Fp::zero(); TEST_N_CHALS].into_boxed_slice())
136                    .try_into()
137                    .unwrap(),
138            ),
139            &PoseidonBN254Parameters,
140            TEST_DOMAIN_SIZE,
141        );
142
143        witness_env
144    }

159    #[test]
160    fn test_regression_ivc_constraints() {
161        let mut constraint_env = ConstraintBuilderEnv::<Fp, IVCLookupTable<Ff1>>::create();
162        constrain_ivc::<Ff1, _>(&mut constraint_env);
163        let constraints = constraint_env.get_relation_constraints();
164
165        let mut constraints_degrees = HashMap::new();
166
167        // Regression testing for the number of constraints and their degree
168        {
169            // 67 + 432 (Poseidon)
170            assert_eq!(constraints.len(), 499);
171            constraints.iter().for_each(|c| {
172                let degree = c.degree(1, 0);
173                *constraints_degrees.entry(degree).or_insert(0) += 1;
174            });
175
176            assert_eq!(constraints_degrees.get(&1), None);
177            assert_eq!(constraints_degrees.get(&2), Some(&6));
178            assert_eq!(constraints_degrees.get(&3), Some(&215));
179            assert_eq!(constraints_degrees.get(&4), Some(&278));
180            assert_eq!(constraints_degrees.get(&5), None);
181
182            // Maximum degree is 5
183            // - fold_iteration increases by one
184            // - the public selectors increase by one
185            assert!(constraints.iter().all(|c| c.degree(1, 0) <= 5));
186        }
187    }
