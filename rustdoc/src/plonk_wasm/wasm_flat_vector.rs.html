1//! The flat vector is a vector of fixed-size elements that we want to expose directly to js-of-ocaml
2//! (so that we can access a `Vec<Field>` cheaply,
3//! by just passing a pointer to a continuous memory region instead of copying.
4//! The wasmvector is a normal heap-allocated vector,
5//! where we leave it on the rust heap and just keep a pointer around.
6//! We use flat for fields, normal for gates etc.
7//!
8//! Accessing Rust vector values is not the same as accessing an array.
9//! Each indexing (e.g. `some_vec[3]`) is costly as it is implemented as a function call.
10//! Knowing that, plus the fact that field elements are implemented as `[u32; 8]`, we know that we incur the cost of following several pointers.
11//! To decrease that cost, we flatten such arrays, going from something like
12//!
13//! ```ignore
14//! [[a0, a1, ..., a7], [b0, b1, ..., b7], ...]
15//! ```
16//!
17//! to a flattened vector like:
18//!
19//! ```ignore
20//! [a0, a1, ..., a7, b0, b1, ..., b7, ...]
21//! ```
22

27#[derive(Clone, Debug)]
28pub struct WasmFlatVector<T>(Vec<T>);

30pub trait FlatVectorElem {
31    const FLATTENED_SIZE: usize;
32    fn flatten(self) -> Vec<u8>;
33    fn unflatten(flat: Vec<u8>) -> Self;
34}

108impl<T: FlatVectorElem> FromWasmAbi for WasmFlatVector<T> {
109    type Abi = <Vec<u8> as FromWasmAbi>::Abi;
110    unsafe fn from_abi(js: Self::Abi) -> Self {
111        let data: Vec<u8> = FromWasmAbi::from_abi(js);
112        let mut res: Vec<T> = Vec::with_capacity(data.len() / T::FLATTENED_SIZE);
113
114        let mut buf = Vec::with_capacity(T::FLATTENED_SIZE);
115        for x in data.into_iter() {
116            assert!(buf.len() < T::FLATTENED_SIZE);
117            buf.push(x);
118            if buf.len() >= T::FLATTENED_SIZE {
119                res.push(T::unflatten(buf));
120                buf = Vec::with_capacity(T::FLATTENED_SIZE);
121            }
122        }
123        assert_eq!(buf.len(), 0);
124        WasmFlatVector(res)
125    }
126}

134impl<T: FlatVectorElem> IntoWasmAbi for WasmFlatVector<T> {
135    type Abi = <Vec<u8> as FromWasmAbi>::Abi;
136    fn into_abi(self) -> Self::Abi {
137        let mut data: Vec<u8> = Vec::with_capacity(self.0.len() * T::FLATTENED_SIZE);
138        for x in self.0.into_iter() {
139            data.extend(x.flatten().into_iter());
140        }
141        IntoWasmAbi::into_abi(data)
142    }
143}
