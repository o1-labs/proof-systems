224#[wasm_bindgen]
225pub fn caml_pasta_fq_to_bytes(x: WasmPastaFq) -> Vec<u8> {
226    let len = core::mem::size_of::<Fq>();
227    let mut str: Vec<u8> = vec![0; len];
228    str.resize(len, 0);
229    let str_as_fq: *mut Fq = str.as_mut_ptr().cast::<Fq>();
230    unsafe {
231        *str_as_fq = x.0;
232    }
233    str
234}

239    if x.len() != len {

243    WasmPastaFq(x)
