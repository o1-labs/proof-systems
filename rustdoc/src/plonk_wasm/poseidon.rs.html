10// fp
11
12#[wasm_bindgen]
13pub fn caml_pasta_fp_poseidon_block_cipher(
14    state: WasmFlatVector<WasmPastaFp>,
15) -> WasmFlatVector<WasmPastaFp> {
16    let mut state_vec: Vec<Fp> = state.into_iter().map(Into::into).collect();
17    poseidon_block_cipher::<Fp, PlonkSpongeConstantsKimchi>(
18        mina_poseidon::pasta::fp_kimchi::static_params(),
19        &mut state_vec,
20    );
21    state_vec
22        .iter()
23        .map(|f| WasmPastaFp(*f))
24        .collect::<Vec<WasmPastaFp>>()
25        .into()
26}
