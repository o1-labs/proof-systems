1use clap::{Parser, ValueEnum};
2use core::str::FromStr;
3use std::{
4    fs::File,
5    io::{self, Write},
6};
7mod vectors;

9#[derive(Debug, Clone, ValueEnum)]
10pub enum Mode {
11    B10,
12    Hex,
13}
14
15impl FromStr for Mode {
16    type Err = ();
17
18    fn from_str(input: &str) -> Result<Self, Self::Err> {
19        match input.to_lowercase().as_str() {
20            "b10" => Ok(Mode::B10),
21            "hex" => Ok(Mode::Hex),
22            _ => Err(()),
23        }
24    }
25}
26
27#[derive(Debug, Clone, ValueEnum)]
28pub enum ParamType {
29    Legacy,
30    Kimchi,
31}
32
33impl FromStr for ParamType {
34    type Err = ();
35
36    fn from_str(input: &str) -> Result<Self, Self::Err> {
37        match input.to_lowercase().as_str() {
38            "legacy" => Ok(ParamType::Legacy),
39            "kimchi" => Ok(ParamType::Kimchi),
40            _ => Err(()),
41        }
42    }
43}
44
45#[derive(Parser)]
46#[command(name = "export_test_vectors")]
47#[command(about = "Export test vectors for the mina-poseidon crate")]
48struct Args {
49    /// Output format for the test vectors
50    #[arg(value_enum)]
51    mode: Mode,
52
53    /// Parameter type to use
54    #[arg(value_enum)]
55    param_type: ParamType,
56
57    /// Output file path, use "-" for stdout
58    output_file: String,
59}
60
61pub fn main() {
62    let args = Args::parse();
63
64    // generate vectors
65    let vectors = vectors::generate(args.mode, args.param_type);
66
67    // save to output file
68    let writer: Box<dyn Write> = match args.output_file.as_str() {
69        "-" => Box::new(io::stdout()),
70        _ => Box::new(File::create(&args.output_file).expect("could not create file")),
71    };
72    serde_json::to_writer_pretty(writer, &vectors).expect("could not write to file");
73}
