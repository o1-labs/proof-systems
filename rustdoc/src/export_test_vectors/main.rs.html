1use clap::{Parser, ValueEnum};
2use core::str::FromStr;
3use std::{
4    fs::File,

7mod vectors;

9#[derive(Debug, Clone, ValueEnum)]
10pub enum Mode {
11    B10,
12    Hex,
13}
14
15impl FromStr for Mode {
16    type Err = ();
17
18    fn from_str(input: &str) -> Result<Self, Self::Err> {
19        match input.to_lowercase().as_str() {
20            "b10" => Ok(Mode::B10),
21            "hex" => Ok(Mode::Hex),
22            _ => Err(()),
23        }
24    }
25}

63#[derive(Parser)]
64#[command(name = "export_test_vectors")]
65#[command(about = "Export test vectors for the mina-poseidon crate")]
66struct Args {
67    /// Number encoding format (base-10 or hexadecimal)
68    #[arg(value_enum)]
69    mode: Mode,
70
71    /// Parameter type to use
72    #[arg(value_enum)]
73    param_type: ParamType,
74
75    /// Output file path, use "-" for stdout
76    output_file: String,
77
78    /// Output file format
79    #[arg(value_enum, default_value = "json", short, long)]
80    format: OutputFormat,
81}
82
83pub fn main() {
84    let args = Args::parse();
85
86    // generate vectors
87    let vectors = vectors::generate(args.mode.clone(), args.param_type.clone());
88
89    // save to output file
90    let mut writer: Box<dyn Write> = match args.output_file.as_str() {
91        "-" => Box::new(io::stdout()),
92        _ => Box::new(File::create(&args.output_file).expect("could not create file")),
93    };
94
95    match args.format {
96        OutputFormat::Es5 => {
97            vectors::write_es5(&mut writer, &vectors, args.param_type)
98                .expect("could not write to file");
99        }
100        OutputFormat::Json => {
101            serde_json::to_writer_pretty(writer, &vectors).expect("could not write to file");
102        }
103    }
104}
