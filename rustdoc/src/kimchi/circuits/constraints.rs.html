234/// Create selector polynomial for a circuit gate
235pub fn selector_polynomial<F: PrimeField>(
236    gate_type: GateType,
237    gates: &[CircuitGate<F>],
238    domain: &EvaluationDomains<F>,
239    target_domain: &D<F>,
240    disable_gates_checks: bool,
241) -> E<F, D<F>> {
242    if cfg!(debug_assertions) && disable_gates_checks {
243        DP::<F>::zero().evaluate_over_domain_by_ref(*target_domain)
244    } else {
245        // Coefficient form
246        let coeff = E::<F, D<F>>::from_vec_and_domain(
247            gates
248                .iter()
249                .map(|gate| {
250                    if gate.typ == gate_type {
251                        F::one()
252                    } else {
253                        F::zero()
254                    }
255                })
256                .collect(),
257            domain.d1,
258        )
259        .interpolate();
260
261        coeff.evaluate_over_domain_by_ref(*target_domain)
262    }
263}

267    /// Returns a [`Builder<F>`]
268    /// It also defaults to the following values of the builder:
269    /// - `public: 0`
270    /// - `prev_challenges: 0`

272    /// - `runtime_tables: None`,
273    /// - `precomputations: None`,
274    /// - `disable_gates_checks: false`,
275    ///
276    /// How to use it:
277    /// 1. Create your instance of your builder for the constraint system using `crate(gates, sponge params)`
278    /// 2. Iterativelly invoke any desired number of steps: `public(), lookup(), runtime(), precomputations()``
279    /// 3. Finally call the `build()` method and unwrap the `Result` to obtain your `ConstraintSystem`

281        Builder {
282            gates,
283            public: 0,
284            prev_challenges: 0,
285            lookup_tables: vec![],
286            runtime_tables: None,
287            precomputations: None,
288            disable_gates_checks: false,
289            max_poly_size: None,
290        }
291    }

320impl<F: PrimeField, G: KimchiCurve<ScalarField = F>, OpeningProof: OpenProof<G>>
321    ProverIndex<G, OpeningProof>
322{
323    /// This function verifies the consistency of the wire
324    /// assignments (witness) against the constraints
325    ///     witness: wire assignment witness
326    ///     RETURN: verification status
327    pub fn verify(&self, witness: &[Vec<F>; COLUMNS], public: &[F]) -> Result<(), GateError> {
328        // pad the witness
329        let pad = vec![F::zero(); self.cs.domain.d1.size() - witness[0].len()];
330        let witness: [Vec<F>; COLUMNS] = array::from_fn(|i| {
331            let mut w = witness[i].to_vec();
332            w.extend_from_slice(&pad);
333            w
334        });
335
336        // check each rows' wiring
337        for (row, gate) in self.cs.gates.iter().enumerate() {
338            // check if wires are connected
339            for col in 0..PERMUTS {
340                let wire = gate.wires[col];
341
342                if wire.col >= PERMUTS {
343                    return Err(GateError::Custom {
344                        row,
345                        err: format!("a wire can only be connected to the first {PERMUTS} columns"),
346                    });
347                }
348
349                if witness[col][row] != witness[wire.col][wire.row] {
350                    return Err(GateError::DisconnectedWires(
351                        Wire { col, row },
352                        Wire {
353                            col: wire.col,
354                            row: wire.row,
355                        },
356                    ));
357                }
358            }
359
360            // for public gates, only the left wire is toggled
361            if row < self.cs.public && gate.coeffs.first() != Some(&F::one()) {
362                return Err(GateError::IncorrectPublic(row));
363            }
364
365            // check the gate's satisfiability
366            gate.verify(row, &witness, self, public)
367                .map_err(|err| GateError::Custom { row, err })?;
368        }
369
370        // all good!
371        Ok(())
372    }
373}

375impl<F: PrimeField> ConstraintSystem<F> {
376    /// evaluate witness polynomials over domains
377    pub fn evaluate(&self, w: &[DP<F>; COLUMNS], z: &DP<F>) -> WitnessOverDomains<F> {
378        // compute shifted witness polynomials and z8, all in parallel
379        let (w8, z8): ([E<F, D<F>>; COLUMNS], _) = {
380            let mut res = w
381                .par_iter()
382                .chain(rayon::iter::once(z))
383                .map(|elem| elem.evaluate_over_domain_by_ref(self.domain.d8))
384                .collect::<Vec<_>>();
385            let z8 = res[COLUMNS].clone();
386            res.truncate(COLUMNS);
387            (res.try_into().unwrap(), z8)
388        };
389
390        let w4: [E<F, D<F>>; COLUMNS] = (0..COLUMNS)
391            .into_par_iter()
392            .map(|i| {
393                E::<F, D<F>>::from_vec_and_domain(
394                    (0..self.domain.d4.size)
395                        .map(|j| w8[i].evals[2 * j as usize])
396                        .collect(),
397                    self.domain.d4,
398                )
399            })
400            .collect::<Vec<_>>()
401            .try_into()
402            .unwrap();
403
404        let z4 = DP::<F>::zero().evaluate_over_domain_by_ref(D::<F>::new(1).unwrap());
405        let z8_shift8 = z8.shift(8);
406
407        let d4_next_w: [_; COLUMNS] = w4
408            .par_iter()
409            .map(|w4_i| w4_i.shift(4))
410            .collect::<Vec<_>>()
411            .try_into()
412            .unwrap();
413
414        let d8_next_w: [_; COLUMNS] = w8
415            .par_iter()
416            .map(|w8_i| w8_i.shift(8))
417            .collect::<Vec<_>>()
418            .try_into()
419            .unwrap();
420
421        WitnessOverDomains {
422            d4: WitnessShifts {
423                next: WitnessEvals {
424                    w: d4_next_w,
425                    // TODO(mimoo): change z to an Option? Or maybe not, we might actually need this dummy evaluation in the aggregated evaluation proof
426                    z: z4.clone(), // dummy evaluation
427                },
428                this: WitnessEvals {
429                    w: w4,
430                    z: z4, // dummy evaluation
431                },
432            },
433            d8: WitnessShifts {
434                next: WitnessEvals {
435                    w: d8_next_w,
436                    z: z8_shift8,
437                },
438                this: WitnessEvals { w: w8, z: z8 },
439            },
440        }
441    }

443    pub(crate) fn evaluated_column_coefficients(&self) -> EvaluatedColumnCoefficients<F> {
444        // compute permutation polynomials
445        let shifts = Shifts::new(&self.domain.d1);
446
447        let n = self.domain.d1.size();
448
449        let mut sigmal1: [Vec<F>; PERMUTS] = array::from_fn(|_| vec![F::zero(); n]);
450
451        for (row, gate) in self.gates.iter().enumerate() {
452            for (cell, sigma) in gate.wires.iter().zip(sigmal1.iter_mut()) {
453                sigma[row] = shifts.cell_to_field(cell);
454            }
455        }
456
457        // Zero out the sigmas in the zk rows, to ensure that the permutation aggregation is
458        // quasi-random for those rows.
459        for row in n + 2 - (self.zk_rows as usize)..n - 1 {
460            for sigma in sigmal1.iter_mut() {
461                sigma[row] = F::zero();
462            }
463        }
464
465        let sigmal1: [_; PERMUTS] = {
466            let [s0, s1, s2, s3, s4, s5, s6] = sigmal1;
467            [
468                E::<F, D<F>>::from_vec_and_domain(s0, self.domain.d1),
469                E::<F, D<F>>::from_vec_and_domain(s1, self.domain.d1),
470                E::<F, D<F>>::from_vec_and_domain(s2, self.domain.d1),
471                E::<F, D<F>>::from_vec_and_domain(s3, self.domain.d1),
472                E::<F, D<F>>::from_vec_and_domain(s4, self.domain.d1),
473                E::<F, D<F>>::from_vec_and_domain(s5, self.domain.d1),
474                E::<F, D<F>>::from_vec_and_domain(s6, self.domain.d1),
475            ]
476        };
477
478        let permutation_coefficients: [DP<F>; PERMUTS] =
479            array::from_fn(|i| sigmal1[i].clone().interpolate());
480
481        // poseidon gate
482        let poseidon_selector = E::<F, D<F>>::from_vec_and_domain(
483            self.gates.iter().map(|gate| gate.ps()).collect(),
484            self.domain.d1,
485        )
486        .interpolate();
487
488        // double generic gate
489        let generic_selector = E::<F, D<F>>::from_vec_and_domain(
490            self.gates
491                .iter()
492                .map(|gate| {
493                    if matches!(gate.typ, GateType::Generic) {
494                        F::one()
495                    } else {
496                        F::zero()
497                    }
498                })
499                .collect(),
500            self.domain.d1,
501        )
502        .interpolate();
503
504        // coefficient polynomial
505        let coefficients: [_; COLUMNS] = array::from_fn(|i| {
506            let padded = self
507                .gates
508                .iter()
509                .map(|gate| gate.coeffs.get(i).cloned().unwrap_or_else(F::zero))
510                .collect();
511            let eval = E::from_vec_and_domain(padded, self.domain.d1);
512            eval.interpolate()
513        });
514
515        EvaluatedColumnCoefficients {
516            permutation_coefficients,
517            coefficients,
518            generic_selector,
519            poseidon_selector,
520        }
521    }

523    pub(crate) fn column_evaluations(
524        &self,
525        evaluated_column_coefficients: &EvaluatedColumnCoefficients<F>,
526    ) -> ColumnEvaluations<F> {
527        let permutation_coefficients8 = array::from_fn(|i| {
528            evaluated_column_coefficients.permutation_coefficients[i]
529                .evaluate_over_domain_by_ref(self.domain.d8)
530        });
531
532        let poseidon_selector8 = evaluated_column_coefficients
533            .poseidon_selector
534            .evaluate_over_domain_by_ref(self.domain.d8);
535
536        // ECC gates
537        let complete_add_selector4 = selector_polynomial(
538            GateType::CompleteAdd,
539            &self.gates,
540            &self.domain,
541            &self.domain.d4,
542            self.disable_gates_checks,
543        );
544
545        let mul_selector8 = selector_polynomial(
546            GateType::VarBaseMul,
547            &self.gates,
548            &self.domain,
549            &self.domain.d8,
550            self.disable_gates_checks,
551        );
552
553        let emul_selector8 = selector_polynomial(
554            GateType::EndoMul,
555            &self.gates,
556            &self.domain,
557            &self.domain.d8,
558            self.disable_gates_checks,
559        );
560
561        let endomul_scalar_selector8 = selector_polynomial(
562            GateType::EndoMulScalar,
563            &self.gates,
564            &self.domain,
565            &self.domain.d8,
566            self.disable_gates_checks,
567        );
568
569        let generic_selector4 = evaluated_column_coefficients
570            .generic_selector
571            .evaluate_over_domain_by_ref(self.domain.d4);
572
573        // RangeCheck0 constraint selector polynomials
574        let range_check0_selector8 = {
575            if !self.feature_flags.range_check0 {
576                None
577            } else {
578                Some(selector_polynomial(
579                    GateType::RangeCheck0,
580                    &self.gates,
581                    &self.domain,
582                    &self.domain.d8,
583                    self.disable_gates_checks,
584                ))
585            }
586        };
587
588        // RangeCheck1 constraint selector polynomials
589        let range_check1_selector8 = {
590            if !self.feature_flags.range_check1 {
591                None
592            } else {
593                Some(selector_polynomial(
594                    GateType::RangeCheck1,
595                    &self.gates,
596                    &self.domain,
597                    &self.domain.d8,
598                    self.disable_gates_checks,
599                ))
600            }
601        };
602
603        // Foreign field addition constraint selector polynomial
604        let foreign_field_add_selector8 = {
605            if !self.feature_flags.foreign_field_add {
606                None
607            } else {
608                Some(selector_polynomial(
609                    GateType::ForeignFieldAdd,
610                    &self.gates,
611                    &self.domain,
612                    &self.domain.d8,
613                    self.disable_gates_checks,
614                ))
615            }
616        };
617
618        // Foreign field multiplication constraint selector polynomial
619        let foreign_field_mul_selector8 = {
620            if !self.feature_flags.foreign_field_mul {
621                None
622            } else {
623                Some(selector_polynomial(
624                    GateType::ForeignFieldMul,
625                    &self.gates,
626                    &self.domain,
627                    &self.domain.d8,
628                    self.disable_gates_checks,
629                ))
630            }
631        };
632
633        let xor_selector8 = {
634            if !self.feature_flags.xor {
635                None
636            } else {
637                Some(selector_polynomial(
638                    GateType::Xor16,
639                    &self.gates,
640                    &self.domain,
641                    &self.domain.d8,
642                    self.disable_gates_checks,
643                ))
644            }
645        };
646
647        let rot_selector8 = {
648            if !self.feature_flags.rot {
649                None
650            } else {
651                Some(selector_polynomial(
652                    GateType::Rot64,
653                    &self.gates,
654                    &self.domain,
655                    &self.domain.d8,
656                    self.disable_gates_checks,
657                ))
658            }
659        };
660
661        // TODO: This doesn't need to be degree 8 but that would require some changes in expr
662        let coefficients8 = array::from_fn(|i| {
663            evaluated_column_coefficients.coefficients[i]
664                .evaluate_over_domain_by_ref(self.domain.d8)
665        });
666
667        ColumnEvaluations {
668            permutation_coefficients8,
669            coefficients8,
670            generic_selector4,
671            poseidon_selector8,
672            complete_add_selector4,
673            mul_selector8,
674            emul_selector8,
675            endomul_scalar_selector8,
676            range_check0_selector8,
677            range_check1_selector8,
678            foreign_field_add_selector8,
679            foreign_field_mul_selector8,
680            xor_selector8,
681            rot_selector8,
682        }
683    }

686/// The default number of chunks in a circuit is one (< 2^16 rows)
687pub const NUM_CHUNKS_BY_DEFAULT: usize = 1;
688
689/// The number of rows required for zero knowledge in circuits with one single chunk
690pub const ZK_ROWS_BY_DEFAULT: u64 = 3;
691
692/// This function computes a strict lower bound in the number of rows required
693/// for zero knowledge in circuits with `num_chunks` chunks. This means that at
694/// least one needs 1 more row than the result of this function to achieve zero
695/// knowledge.
696/// Example:
697///   for 1 chunk, this function returns 2, but at least 3 rows are needed
698/// Note:
699///   the number of zero knowledge rows is usually computed across the codebase
700///   as the formula `(16 * num_chunks + 5) / 7`, which is precisely the formula
701///   in this function plus one.
702pub fn zk_rows_strict_lower_bound(num_chunks: usize) -> usize {
703    (2 * (PERMUTS + 1) * num_chunks - 2) / PERMUTS
704}

706impl FeatureFlags {
707    pub fn from_gates_and_lookup_features<F: PrimeField>(
708        gates: &[CircuitGate<F>],
709        lookup_features: LookupFeatures,
710    ) -> FeatureFlags {
711        let mut feature_flags = FeatureFlags {
712            range_check0: false,
713            range_check1: false,
714            lookup_features,
715            foreign_field_add: false,
716            foreign_field_mul: false,
717            xor: false,
718            rot: false,
719        };
720
721        for gate in gates {
722            match gate.typ {
723                GateType::RangeCheck0 => feature_flags.range_check0 = true,
724                GateType::RangeCheck1 => feature_flags.range_check1 = true,
725                GateType::ForeignFieldAdd => feature_flags.foreign_field_add = true,
726                GateType::ForeignFieldMul => feature_flags.foreign_field_mul = true,
727                GateType::Xor16 => feature_flags.xor = true,
728                GateType::Rot64 => feature_flags.rot = true,
729                _ => (),
730            }
731        }
732
733        feature_flags
734    }

808    /// Build the [ConstraintSystem] from a [Builder].
809    pub fn build(self) -> Result<ConstraintSystem<F>, SetupError> {
810        let mut gates = self.gates;
811        let lookup_tables = self.lookup_tables;
812        let runtime_tables = self.runtime_tables;
813
814        //~ 1. If the circuit is less than 2 gates, abort.
815        // for some reason we need more than 1 gate for the circuit to work, see TODO below
816        assert!(gates.len() > 1);
817
818        let feature_flags = FeatureFlags::from_gates(&gates, runtime_tables.is_some());
819
820        let lookup_domain_size = {
821            // First we sum over the lookup table size
822            let mut has_table_with_id_0 = false;
823            let mut lookup_domain_size: usize = lookup_tables
824                .iter()
825                .map(|LookupTable { id, data }| {
826                    // See below for the reason
827                    if *id == 0_i32 {
828                        has_table_with_id_0 = true
829                    }
830                    if data.is_empty() {
831                        0
832                    } else {
833                        data[0].len()
834                    }
835                })
836                .sum();
837            // After that on the runtime tables
838            if let Some(runtime_tables) = runtime_tables.as_ref() {
839                // FIXME: Check that a runtime table with ID 0 is enforced to
840                // contain a zero entry row.
841                for runtime_table in runtime_tables.iter() {
842                    lookup_domain_size += runtime_table.len();
843                }
844            }
845            // And we add the built-in tables, depending on the features.
846            let LookupFeatures { patterns, .. } = &feature_flags.lookup_features;
847            let mut gate_lookup_tables = GateLookupTables {
848                xor: false,
849                range_check: false,
850            };
851            for pattern in patterns.into_iter() {
852                if let Some(gate_table) = pattern.table() {
853                    gate_lookup_tables[gate_table] = true
854                }
855            }
856            for gate_table in gate_lookup_tables.into_iter() {
857                lookup_domain_size += gate_table.table_size();
858            }
859
860            // A dummy zero entry will be added if there is no table with ID
861            // zero. Therefore we must count this in the size.
862            if has_table_with_id_0 {
863                lookup_domain_size
864            } else {
865                lookup_domain_size + 1
866            }
867        };
868
869        //~ 1. Compute the number of zero-knowledge rows (`zk_rows`) that will be required to
870        //~    achieve zero-knowledge. The following constraints apply to `zk_rows`:
871        //~    * The number of chunks `c` results in an evaluation at `zeta` and `zeta * omega` in
872        //~      each column for `2*c` evaluations per column, so `zk_rows >= 2*c + 1`.
873        //~    * The permutation argument interacts with the `c` chunks in parallel, so it is
874        //~      possible to cross-correlate between them to compromise zero knowledge. We know
875        //~      that there is some `c >= 1` such that `zk_rows = 2*c + k` from the above. Thus,
876        //~      attempting to find the evaluation at a new point, we find that:
877        //~      * the evaluation of every witness column in the permutation contains `k` unknowns;
878        //~      * the evaluations of the permutation argument aggregation has `k-1` unknowns;
879        //~      * the permutation argument applies on all but `zk_rows - 3` rows;
880        //~      * and thus we form the equation `zk_rows - 3 < 7 * k + (k - 1)` to ensure that we
881        //~        can construct fewer equations than we have unknowns.
882        //~
883        //~    This simplifies to `k > (2 * c - 2) / 7`, giving `zk_rows > (16 * c - 2) / 7`.
884        //~    We can derive `c` from the `max_poly_size` supported by the URS, and thus we find
885        //~    `zk_rows` and `domain_size` satisfying the fixpoint
886        //~
887        //~    ```text
888        //~    zk_rows = (16 * (domain_size / max_poly_size) + 5) / 7
889        //~    domain_size = circuit_size + zk_rows
890        //~    ```
891        //~
892        let (zk_rows, domain_size_lower_bound) = {
893            // We add 1 to the lookup domain size because there is one element
894            // used to close the permutation argument (the polynomial Z is of
895            // degree n + 1 where n is the order of the subgroup H).
896            let circuit_lower_bound = core::cmp::max(gates.len(), lookup_domain_size + 1);
897            let get_domain_size_lower_bound = |zk_rows: u64| circuit_lower_bound + zk_rows as usize;
898
899            let mut zk_rows = 3;
900            let mut domain_size_lower_bound = get_domain_size_lower_bound(zk_rows);
901            if let Some(max_poly_size) = self.max_poly_size {
902                // Iterate to find a fixed-point where zk_rows is sufficient for the number of
903                // chunks that we use, and also does not cause us to overflow the domain size.
904                // NB: We use iteration here rather than hard-coding an assumption about
905                // `compute_size_of_domain`s internals. In practice, this will never be executed
906                // more than once.
907                while {
908                    let domain_size = D::<F>::compute_size_of_domain(domain_size_lower_bound)
909                        .ok_or(SetupError::DomainCreation(
910                            DomainCreationError::DomainSizeFailed(domain_size_lower_bound),
911                        ))?;
912                    let num_chunks = if domain_size < max_poly_size {
913                        1
914                    } else {
915                        domain_size / max_poly_size
916                    };
917                    zk_rows = (zk_rows_strict_lower_bound(num_chunks) + 1) as u64;
918                    domain_size_lower_bound = get_domain_size_lower_bound(zk_rows);
919                    domain_size < domain_size_lower_bound
920                } {}
921            }
922            (zk_rows, domain_size_lower_bound)
923        };
924
925        //~ 1. Create a domain for the circuit. That is,
926        //~    compute the smallest subgroup of the field that
927        //~    has order greater or equal to `n + zk_rows` elements.
928        let domain = EvaluationDomains::<F>::create(domain_size_lower_bound)
929            .map_err(SetupError::DomainCreation)?;
930
931        assert!(domain.d1.size > zk_rows);
932
933        //~ 1. Pad the circuit: add zero gates to reach the domain size.
934        let d1_size = domain.d1.size();
935        let mut padding = (gates.len()..d1_size)
936            .map(|i| {
937                CircuitGate::<F>::zero(array::from_fn(|j| Wire {
938                    col: WIRES[j],
939                    row: i,
940                }))
941            })
942            .collect();
943        gates.append(&mut padding);
944
945        //~ 1. sample the `PERMUTS` shifts.
946        let shifts = Shifts::new(&domain.d1);
947
948        //
949        // Lookup
950        // ------
951        let lookup_constraint_system = LookupConstraintSystem::create(
952            &gates,
953            lookup_tables,
954            runtime_tables,
955            &domain,
956            zk_rows as usize,
957        )
958        .map_err(SetupError::LookupCreation)?;
959
960        let sid = shifts.map[0].clone();
961
962        // TODO: remove endo as a field
963        let endo = F::zero();
964
965        let domain_constant_evaluation = OnceCell::new();
966
967        let constraints = ConstraintSystem {
968            domain,
969            public: self.public,
970            prev_challenges: self.prev_challenges,
971            sid,
972            gates,
973            shift: shifts.shifts,
974            endo,
975            zk_rows,
976            //fr_sponge_params: self.sponge_params,
977            lookup_constraint_system,
978            feature_flags,
979            precomputations: domain_constant_evaluation,
980            disable_gates_checks: self.disable_gates_checks,
981        };
982
983        match self.precomputations {
984            Some(t) => {
985                constraints.set_precomputations(t);
986            }
987            None => {
988                constraints.precomputations();
989            }
990        }
991        Ok(constraints)
992    }
