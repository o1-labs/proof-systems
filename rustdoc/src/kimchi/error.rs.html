7/// Errors that can arise when creating a proof
8// TODO(mimoo): move this out of oracle
9#[derive(Error, Debug, Clone, Copy)]
10pub enum ProverError {
11    #[error("the circuit is too large")]
12    NoRoomForZkInWitness,
13
14    #[error(
15        "there are not enough random rows to achieve zero-knowledge (expected: {0}, got: {1})"
16    )]
17    NotZeroKnowledge(usize, usize),
18
19    #[error("the witness columns are not all the same size")]
20    WitnessCsInconsistent,
21
22    #[error("the proof could not be constructed: {0}")]
23    Prover(&'static str),
24
25    #[error("the permutation was not constructed correctly: {0}")]
26    Permutation(&'static str),
27
28    #[error("the lookup failed to find a match in the table: row={0}")]
29    ValueNotInTable(usize),
30
31    #[error("the runtime tables provided did not match the index's configuration")]
32    RuntimeTablesInconsistent,
33
34    #[error("wrong number of custom blinders given: {0}")]
35    WrongBlinders(CommitmentError),
36}

38/// Errors that can arise when verifying a proof
39#[derive(Error, Debug, Clone, Copy)]
40pub enum VerifyError {
41    #[error("the commitment to {0} is of an unexpected size (expected {1}, got {2})")]
42    IncorrectCommitmentLength(&'static str, usize, usize),
43
44    #[error("the public input is of an unexpected size (expected {0})")]
45    IncorrectPubicInputLength(usize),
46
47    #[error("the previous challenges have an unexpected length (expected {0}, got {1})")]
48    IncorrectPrevChallengesLength(usize, usize),
49
50    #[error(
51        "proof malformed: an evaluation for {2} was of the incorrect size (expected {0}, got {1})"
52    )]
53    IncorrectEvaluationsLength(usize, usize, &'static str),
54
55    #[error("the opening proof failed to verify")]
56    OpenProof,
57
58    #[error("lookup used in circuit, but proof is missing lookup commitments")]
59    LookupCommitmentMissing,
60
61    #[error("lookup used in circuit, but proof is missing lookup evaluations")]
62    LookupEvalsMissing,
63
64    #[error("lookup used in circuit, but proof has inconsistent number of lookup evaluations and commitments")]
65    ProofInconsistentLookup,
66
67    #[error("cannot batch proofs using different SRSes")]
68    DifferentSRS,
69
70    #[error("SRS size is smaller than the domain size required by the circuit")]
71    SRSTooSmall,
72
73    #[error("runtime tables are used, but missing from the proof")]
74    IncorrectRuntimeProof,
75
76    #[error("the evaluation for {0:?} is missing")]
77    MissingEvaluation(crate::circuits::berkeley_columns::Column),
78
79    #[error("the evaluation for PublicInput is missing")]
80    MissingPublicInputEvaluation,
81
82    #[error("the commitment for {0:?} is missing")]
83    MissingCommitment(crate::circuits::berkeley_columns::Column),
84}
