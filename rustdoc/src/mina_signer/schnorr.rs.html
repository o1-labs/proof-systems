28/// Schnorr signer context for the Mina signature algorithm
29///
30/// For details about the signature algorithm please see the
31/// [`schnorr`](crate::schnorr) documentation
32pub struct Schnorr<H: Hashable> {
33    hasher: Box<dyn Hasher<Message<H>>>,
34    domain_param: H::D,
35}
36
37#[derive(Clone)]
38struct Message<H: Hashable> {
39    input: H,
40    pub_key_x: BaseField,
41    pub_key_y: BaseField,
42    rx: BaseField,
43}
44
45impl<H: Hashable> Hashable for Message<H> {
46    type D = H::D;
47
48    fn to_roinput(&self) -> ROInput {
49        self.input
50            .to_roinput()
51            .append_field(self.pub_key_x)
52            .append_field(self.pub_key_y)
53            .append_field(self.rx)
54    }
55
56    fn domain_string(domain_param: Self::D) -> Option<String> {
57        H::domain_string(domain_param)
58    }
59}

61impl<H: 'static + Hashable> Signer<H> for Schnorr<H> {
62    fn sign(&mut self, kp: &Keypair, input: &H) -> Signature {
63        let k: ScalarField = self.derive_nonce(kp, input);
64        let r: CurvePoint = CurvePoint::generator()
65            .mul_bigint(k.into_bigint())
66            .into_affine();
67        let k: ScalarField = if r.y.into_bigint().is_even() { k } else { -k };
68
69        let e: ScalarField = self.message_hash(&kp.public, r.x, input);
70        let s: ScalarField = k + e * kp.secret.scalar();
71
72        Signature::new(r.x, s)
73    }
74
75    fn verify(&mut self, sig: &Signature, public: &PubKey, input: &H) -> bool {
76        let ev: ScalarField = self.message_hash(public, sig.rx, input);
77
78        let sv = CurvePoint::generator()
79            .mul_bigint(sig.s.into_bigint())
80            .into_affine();
81        // Perform addition and infinity check in projective coordinates for
82        // performance
83        let rv = public.point().mul_bigint(ev.into_bigint()).neg().add(sv);
84
85        if rv.is_zero() {
86            return false;
87        }
88
89        let rv = rv.into_affine();
90
91        rv.y.into_bigint().is_even() && rv.x == sig.rx
92    }
93}

113impl<H: 'static + Hashable> Schnorr<H> {
114    /// This function uses a cryptographic hash function to create a uniformly
115    /// and randomly distributed nonce. It is crucial for security that no two
116    /// different messages share the same nonce.
117    fn derive_nonce(&self, kp: &Keypair, input: &H) -> ScalarField {
118        let mut blake_hasher = Blake2bVar::new(32).unwrap();
119
120        let roi = input
121            .to_roinput()
122            .append_field(kp.public.point().x)
123            .append_field(kp.public.point().y)
124            .append_scalar(*kp.secret.scalar())
125            .append_bytes(&self.domain_param.clone().into_bytes());
126
127        blake_hasher.update(&roi.to_bytes());
128
129        let mut bytes = [0; 32];
130        blake_hasher
131            .finalize_variable(&mut bytes)
132            .expect("incorrect output size");
133        // Drop the top two bits to convert into a scalar field element
134        //   N.B. Since the order of Pallas's scalar field p is very close to 2^m
135        //   for some m, truncating only creates a tiny amount of bias that should
136        //   be insignificant and better than reduction modulo p.
137        bytes[bytes.len() - 1] &= 0b0011_1111;
138
139        ScalarField::from_random_bytes(&bytes[..]).expect("failed to create scalar from bytes")
140    }
141
142    /// This function uses a cryptographic hash function (based on a sponge
143    /// construction) to convert the message to be signed (and some other
144    /// information) into a uniformly and randomly distributed scalar field
145    /// element. It uses Mina's variant of the Poseidon SNARK-friendly
146    /// cryptographic hash function.
147    /// Details:
148    /// <https://github.com/o1-labs/cryptography-rfcs/blob/httpsnapps-notary-signatures/mina/001-poseidon-sponge.md>
149    fn message_hash(&mut self, pub_key: &PubKey, rx: BaseField, input: &H) -> ScalarField {
150        let schnorr_input = Message::<H> {
151            input: input.clone(),
152            pub_key_x: pub_key.point().x,
153            pub_key_y: pub_key.point().y,
154            rx,
155        };
156
157        // Squeeze and convert from base field element to scalar field element
158        // Since the difference in modulus between the two fields is < 2^125,
159        // w.h.p., a random value from one field will fit in the other field.
160        ScalarField::from(self.hasher.hash(&schnorr_input).into_bigint())
161    }
162}
