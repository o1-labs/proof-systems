48#[ocaml_gen::func]
49#[ocaml::func]
50pub fn caml_pasta_fq_plonk_index_create(
51    gates: CamlPastaFqPlonkGateVectorPtr,
52    public: ocaml::Int,
53    lookup_tables: Vec<CamlLookupTable<CamlFq>>,
54    runtime_tables: Vec<CamlRuntimeTableCfg<CamlFq>>,
55    prev_challenges: ocaml::Int,
56    srs: CamlFqSrs,
57    lazy_mode: bool,
58) -> Result<CamlPastaFqPlonkIndex, ocaml::Error> {
59    let gates: Vec<_> = gates
60        .as_ref()
61        .0
62        .iter()
63        .map(|gate| CircuitGate::<Fq> {
64            typ: gate.typ,
65            wires: gate.wires,
66            coeffs: gate.coeffs.clone(),
67        })
68        .collect();
69
70    let runtime_tables: Vec<RuntimeTableCfg<Fq>> =
71        runtime_tables.into_iter().map(Into::into).collect();
72
73    let lookup_tables: Vec<LookupTable<Fq>> = lookup_tables.into_iter().map(Into::into).collect();
74
75    // create constraint system
76    let cs = ConstraintSystem::<Fq>::create(gates)
77        .public(public as usize)
78        .prev_challenges(prev_challenges as usize)
79        .lookup(lookup_tables)
80        .runtime(if runtime_tables.is_empty() {
81            None
82        } else {
83            Some(runtime_tables)
84        })
85        .lazy_mode(lazy_mode)
86        .build()?;
87
88    // endo
89    let (endo_q, _endo_r) = poly_commitment::ipa::endos::<Vesta>();
90
91    srs.0.with_lagrange_basis(cs.domain.d1);
92
93    // create index
94    let mut index =
95        ProverIndex::<Pallas, OpeningProof<Pallas>>::create(cs, endo_q, srs.clone(), lazy_mode);
96    // Compute and cache the verifier index digest
97    index.compute_verifier_index_digest::<DefaultFqSponge<PallasParameters, PlonkSpongeConstantsKimchi>>();
98
99    Ok(CamlPastaFqPlonkIndex(Box::new(index)))
100}

132#[ocaml_gen::func]
133#[ocaml::func]
134pub fn caml_pasta_fq_plonk_index_read(
135    offset: Option<ocaml::Int>,
136    srs: CamlFqSrs,
137    path: String,
138) -> Result<CamlPastaFqPlonkIndex, ocaml::Error> {
139    // open the file for reading
140    let file = match File::open(path) {
141        Err(_) => {
142            return Err(
143                ocaml::Error::invalid_argument("caml_pasta_fp_plonk_index_read")
144                    .err()
145                    .unwrap(),
146            )
147        }
148        Ok(file) => file,
149    };
150    let mut r = BufReader::new(file);
151
152    // optional offset in file
153    if let Some(offset) = offset {
154        r.seek(Start(offset as u64))?;
155    }
156
157    // deserialize the index
158    let mut t = ProverIndex::<Pallas, OpeningProof<Pallas>>::deserialize(
159        &mut rmp_serde::Deserializer::new(r),
160    )?;
161    t.srs = srs.clone();
162
163    let (linearization, powers_of_alpha) = expr_linearization(Some(&t.cs.feature_flags), true);
164    t.linearization = linearization;
165    t.powers_of_alpha = powers_of_alpha;
166
167    Ok(CamlPastaFqPlonkIndex(Box::new(t)))
168}

171#[ocaml::func]
172pub fn caml_pasta_fq_plonk_index_write(
173    append: Option<bool>,
174    index: CamlPastaFqPlonkIndexPtr<'static>,
175    path: String,
176) -> Result<(), ocaml::Error> {
177    let file = OpenOptions::new()
178        .append(append.unwrap_or(true))
179        .open(path)
180        .map_err(|_| {
181            ocaml::Error::invalid_argument("caml_pasta_fq_plonk_index_write")
182                .err()
183                .unwrap()
184        })?;
185    let w = BufWriter::new(file);
186    index
187        .as_ref()
188        .0
189        .serialize(&mut rmp_serde::Serializer::new(w))
190        .map_err(|e| e.into())
191}
