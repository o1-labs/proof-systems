27/// Creates an iterator that repeats an element `n` times.
28/// This function is stable in Rust 1.82+ (lint exists since Rust 1.83+).
29/// We keep a manual implementation for compatibility with older Rust versions.
30/// TODO: Remove when updating to Rust 1.85+. See <https://github.com/o1-labs/mina-rust/issues/1951>
31#[rustversion::attr(since(1.83), allow(clippy::manual_repeat_n))]
32pub fn repeat_n<T: Clone>(element: T, count: usize) -> impl Iterator<Item = T> {
33    core::iter::repeat(element).take(count)
34}

36/// Check if `a` is a multiple of `b`.
37/// This method is stable in Rust 1.85+.
38/// We keep a manual implementation for compatibility with older Rust versions.
39/// TODO: Remove when updating to Rust 1.85+. See <https://github.com/o1-labs/mina-rust/issues/1951>
40#[rustversion::attr(since(1.85), allow(clippy::manual_is_multiple_of))]
41pub fn is_multiple_of<T>(a: T, b: T) -> bool
42where
43    T: core::ops::Rem<Output = T> + PartialEq + From<u8>,
44{
45    a % b == T::from(0u8)
46}
