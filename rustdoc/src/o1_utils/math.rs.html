3/// Returns ceil(log2(d)) but panics if d = 0.
4pub fn ceil_log2(d: usize) -> usize {
5    // NOTE: should this really be usize, since usize is depended on the underlying system architecture?
6
7    assert!(d != 0);
8    let mut pow2 = 1;
9    let mut ceil_log2 = 0;
10    while d > pow2 {
11        ceil_log2 += 1;
12        pow2 = match pow2.checked_mul(2) {
13            Some(x) => x,
14            None => break,
15        }
16    }
17    ceil_log2
18}

20/// Integer division rounding up.
21/// This function is now stable in Rust 1.73+. See <https://github.com/rust-lang/rust/issues/88581>
22/// We keep a manual implementation for compatibility with older Rust versions.
23/// TODO: Remove when updating to Rust 1.85+. See <https://github.com/o1-labs/mina-rust/issues/1951>
24#[rustversion::attr(since(1.85), allow(clippy::manual_div_ceil))]
25pub const fn div_ceil(a: usize, b: usize) -> usize {
26    (a + b - 1) / b
27}
